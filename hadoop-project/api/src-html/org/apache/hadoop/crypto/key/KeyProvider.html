037/**
038 * A provider of secret key material for Hadoop applications. Provides an
039 * abstraction to separate key storage from users of encryption. It
040 * is intended to support getting or storing keys in a variety of ways,
041 * including third party bindings.
042 */
043@InterfaceAudience.Public
044@InterfaceStability.Unstable
045public abstract class KeyProvider {
046  public static final String DEFAULT_CIPHER_NAME =
047      "hadoop.security.key.default.cipher";
048  public static final String DEFAULT_CIPHER = "AES/CTR/NoPadding";
049  public static final String DEFAULT_BITLENGTH_NAME =
050      "hadoop.security.key.default.bitlength";
051  public static final int DEFAULT_BITLENGTH = 256;
052
053  /**
054   * The combination of both the key version name and the key material.
055   */
056  public static class KeyVersion {
057    private final String versionName;
058    private final byte[] material;
059
060    protected KeyVersion(String versionName,
061                         byte[] material) {
062      this.versionName = versionName;
063      this.material = material;
064    }
065
066    public String getVersionName() {
067      return versionName;
068    }
069
070    public byte[] getMaterial() {
071      return material;
072    }
073
074    public String toString() {
075      StringBuilder buf = new StringBuilder();
076      buf.append("key(");
077      buf.append(versionName);
078      buf.append(")=");
079      if (material == null) {
080        buf.append("null");
081      } else {
082        for(byte b: material) {
083          buf.append(' ');
084          int right = b & 0xff;
085          if (right < 0x10) {
086            buf.append('0');
087          }
088          buf.append(Integer.toHexString(right));
089        }
090      }
091      return buf.toString();
092    }
093  }

095  /**
096   * Key metadata that is associated with the key.
097   */
098  public static class Metadata {
099    private final static String CIPHER_FIELD = "cipher";
100    private final static String BIT_LENGTH_FIELD = "bitLength";
101    private final static String CREATED_FIELD = "created";
102    private final static String VERSIONS_FIELD = "versions";
103
104    private final String cipher;
105    private final int bitLength;
106    private final Date created;
107    private int versions;
108
109    protected Metadata(String cipher, int bitLength,
110                       Date created, int versions) {
111      this.cipher = cipher;
112      this.bitLength = bitLength;
113      this.created = created;
114      this.versions = versions;
115    }
116
117    public Date getCreated() {
118      return created;
119    }
120
121    public String getCipher() {
122      return cipher;
123    }
124
125    /**
126     * Get the algorithm from the cipher.
127     * @return the algorithm name
128     */
129    public String getAlgorithm() {
130      int slash = cipher.indexOf('/');
131      if (slash == - 1) {
132        return cipher;
133      } else {
134        return cipher.substring(0, slash);
135      }
136    }
137
138    public int getBitLength() {
139      return bitLength;
140    }
141
142    public int getVersions() {
143      return versions;
144    }
145
146    protected int addVersion() {
147      return versions++;
148    }
149
150    /**
151     * Serialize the metadata to a set of bytes.
152     * @return the serialized bytes
153     * @throws IOException
154     */
155    protected byte[] serialize() throws IOException {
156      ByteArrayOutputStream buffer = new ByteArrayOutputStream();
157      JsonWriter writer = new JsonWriter(new OutputStreamWriter(buffer));
158      writer.beginObject();
159      if (cipher != null) {
160        writer.name(CIPHER_FIELD).value(cipher);
161      }
162      if (bitLength != 0) {
163        writer.name(BIT_LENGTH_FIELD).value(bitLength);
164      }
165      if (created != null) {
166        writer.name(CREATED_FIELD).value(created.getTime());
167      }
168      writer.name(VERSIONS_FIELD).value(versions);
169      writer.endObject();
170      writer.flush();
171      return buffer.toByteArray();
172    }
173
174    /**
175     * Deserialize a new metadata object from a set of bytes.
176     * @param bytes the serialized metadata
177     * @throws IOException
178     */
179    protected Metadata(byte[] bytes) throws IOException {
180      String cipher = null;
181      int bitLength = 0;
182      Date created = null;
183      int versions = 0;
184      JsonReader reader = new JsonReader(new InputStreamReader
185          (new ByteArrayInputStream(bytes)));
186      reader.beginObject();
187      while (reader.hasNext()) {
188        String field = reader.nextName();
189        if (CIPHER_FIELD.equals(field)) {
190          cipher = reader.nextString();
191        } else if (BIT_LENGTH_FIELD.equals(field)) {
192          bitLength = reader.nextInt();
193        } else if (CREATED_FIELD.equals(field)) {
194          created = new Date(reader.nextLong());
195        } else if (VERSIONS_FIELD.equals(field)) {
196          versions = reader.nextInt();
197        }
198      }
199      reader.endObject();
200      this.cipher = cipher;
201      this.bitLength = bitLength;
202      this.created = created;
203      this.versions = versions;
204    }
205  }

207  /**
208   * Options when creating key objects.
209   */
210  public static class Options {
211    private String cipher;
212    private int bitLength;
213
214    public Options(Configuration conf) {
215      cipher = conf.get(DEFAULT_CIPHER_NAME, DEFAULT_CIPHER);
216      bitLength = conf.getInt(DEFAULT_BITLENGTH_NAME, DEFAULT_BITLENGTH);
217    }
218
219    public Options setCipher(String cipher) {
220      this.cipher = cipher;
221      return this;
222    }
223
224    public Options setBitLength(int bitLength) {
225      this.bitLength = bitLength;
226      return this;
227    }
228
229    protected String getCipher() {
230      return cipher;
231    }
232
233    protected int getBitLength() {
234      return bitLength;
235    }
236  }

282  /**
283   * Get the current version of the key, which should be used for encrypting new
284   * data.
285   * @param name the base name of the key
286   * @return the version name of the current version of the key or null if the
287   *    key version doesn't exist
288   * @throws IOException
289   */
290  public KeyVersion getCurrentKey(String name) throws IOException {
291    Metadata meta = getMetadata(name);
292    if (meta == null) {
293      return null;
294    }
295    return getKeyVersion(buildVersionName(name, meta.getVersions() - 1));
296  }

341  /**
342   * Split the versionName in to a base name. Converts "/aaa/bbb/3" to
343   * "/aaa/bbb".
344   * @param versionName the version name to split
345   * @return the base name of the key
346   * @throws IOException
347   */
348  public static String getBaseName(String versionName) throws IOException {
349    int div = versionName.lastIndexOf('@');
350    if (div == -1) {
351      throw new IOException("No version in key path " + versionName);
352    }
353    return versionName.substring(0, div);
354  }
355
356  /**
357   * Build a version string from a basename and version number. Converts
358   * "/aaa/bbb" and 3 to "/aaa/bbb@3".
359   * @param name the basename of the key
360   * @param version the version of the key
361   * @return the versionName of the key.
362   */
363  protected static String buildVersionName(String name, int version) {
364    return name + "@" + version;
365  }

367  /**
368   * Convert a nested URI to decode the underlying path. The translation takes
369   * the authority and parses it into the underlying scheme and authority.
370   * For example, "myscheme://hdfs@nn/my/path" is converted to
371   * "hdfs://nn/my/path".
372   * @param nestedUri the URI from the nested URI
373   * @return the unnested path
374   */
375  public static Path unnestUri(URI nestedUri) {
376    String[] parts = nestedUri.getAuthority().split("@", 2);
377    StringBuilder result = new StringBuilder(parts[0]);
378    result.append("://");
379    if (parts.length == 2) {
380      result.append(parts[1]);
381    }
382    result.append(nestedUri.getPath());
383    if (nestedUri.getQuery() != null) {
384      result.append("?");
385      result.append(nestedUri.getQuery());
386    }
387    if (nestedUri.getFragment() != null) {
388      result.append("#");
389      result.append(nestedUri.getFragment());
390    }
391    return new Path(result.toString());
392  }

394  /**
395   * Find the provider with the given key.
396   * @param providerList the list of providers
397   * @param keyName the key name we are looking for
398   * @return the KeyProvider that has the key
399   */
400  public static KeyProvider findProvider(List<KeyProvider> providerList,
401                                         String keyName) throws IOException {
402    for(KeyProvider provider: providerList) {
403      if (provider.getMetadata(keyName) != null) {
404        return provider;
405      }
406    }
407    throw new IOException("Can't find KeyProvider for key " + keyName);
408  }
