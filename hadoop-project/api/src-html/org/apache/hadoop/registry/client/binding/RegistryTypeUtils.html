019package org.apache.hadoop.registry.client.binding;
020
021import com.google.common.base.Preconditions;
022import org.apache.hadoop.classification.InterfaceAudience;
023import org.apache.hadoop.classification.InterfaceStability;
024import org.apache.hadoop.registry.client.exceptions.InvalidRecordException;
025import org.apache.hadoop.registry.client.types.AddressTypes;
026import org.apache.hadoop.registry.client.types.Endpoint;
027import org.apache.hadoop.registry.client.types.ProtocolTypes;
028
029import java.net.InetSocketAddress;
030import java.net.MalformedURLException;
031import java.net.URI;
032import java.net.URL;
033import java.util.ArrayList;
034import java.util.Arrays;
035import java.util.List;
036
037/**
038 * Static methods to work with registry types â€”primarily endpoints and the
039 * list representation of addresses.
040 */
041@InterfaceAudience.Public
042@InterfaceStability.Evolving
043public class RegistryTypeUtils {
044
045  /**
046   * Create a URL endpoint from a list of URIs
047   * @param api implemented API
048   * @param protocolType protocol type
049   * @param uris URIs
050   * @return a new endpoint
051   */
052  public static Endpoint urlEndpoint(String api,
053      String protocolType,
054      URI... uris) {
055    return new Endpoint(api, protocolType, uris);
056  }
057
058  /**
059   * Create a REST endpoint from a list of URIs
060   * @param api implemented API
061   * @param uris URIs
062   * @return a new endpoint
063   */
064  public static Endpoint restEndpoint(String api,
065      URI... uris) {
066    return urlEndpoint(api, ProtocolTypes.PROTOCOL_REST, uris);
067  }
068
069  /**
070   * Create a Web UI endpoint from a list of URIs
071   * @param api implemented API
072   * @param uris URIs
073   * @return a new endpoint
074   */
075  public static Endpoint webEndpoint(String api,
076      URI... uris) {
077    return urlEndpoint(api, ProtocolTypes.PROTOCOL_WEBUI, uris);
078  }
079
080  /**
081   * Create an internet address endpoint from a list of URIs
082   * @param api implemented API
083   * @param protocolType protocol type
084   * @param hostname hostname/FQDN
085   * @param port port
086   * @return a new endpoint
087   */
088
089  public static Endpoint inetAddrEndpoint(String api,
090      String protocolType,
091      String hostname,
092      int port) {
093    Preconditions.checkArgument(api != null, "null API");
094    Preconditions.checkArgument(protocolType != null, "null protocolType");
095    Preconditions.checkArgument(hostname != null, "null hostname");
096    return new Endpoint(api,
097        AddressTypes.ADDRESS_HOSTNAME_AND_PORT,
098        protocolType,
099        tuplelist(hostname, Integer.toString(port)));
100  }
101
102  /**
103   * Create an IPC endpoint
104   * @param api API
105   * @param protobuf flag to indicate whether or not the IPC uses protocol
106   * buffers
107   * @param address the address as a tuple of (hostname, port)
108   * @return the new endpoint
109   */
110  public static Endpoint ipcEndpoint(String api,
111      boolean protobuf, List<String> address) {
112    ArrayList<List<String>> addressList = new ArrayList<List<String>>();
113    if (address != null) {
114      addressList.add(address);
115    }
116    return new Endpoint(api,
117        AddressTypes.ADDRESS_HOSTNAME_AND_PORT,
118        protobuf ? ProtocolTypes.PROTOCOL_HADOOP_IPC_PROTOBUF
119                 : ProtocolTypes.PROTOCOL_HADOOP_IPC,
120        addressList);
121  }
122
123  /**
124   * Create a single-element list of tuples from the input.
125   * that is, an input ("a","b","c") is converted into a list
126   * in the form [["a","b","c"]]
127   * @param t1 tuple elements
128   * @return a list containing a single tuple
129   */
130  public static List<List<String>> tuplelist(String... t1) {
131    List<List<String>> outer = new ArrayList<List<String>>();
132    outer.add(tuple(t1));
133    return outer;
134  }
135
136  /**
137   * Create a tuples from the input.
138   * that is, an input ("a","b","c") is converted into a list
139   * in the form ["a","b","c"]
140   * @param t1 tuple elements
141   * @return a single tuple as a list
142   */
143  public static List<String> tuple(String... t1) {
144    return Arrays.asList(t1);
145  }
146
147  /**
148   * Create a tuples from the input, converting all to Strings in the process
149   * that is, an input ("a", 7, true) is converted into a list
150   * in the form ["a","7,"true"]
151   * @param t1 tuple elements
152   * @return a single tuple as a list
153   */
154  public static List<String> tuple(Object... t1) {
155    List<String> l = new ArrayList<String>(t1.length);
156    for (Object t : t1) {
157      l.add(t.toString());
158    }
159    return l;
160  }
161
162  /**
163   * Convert a socket address pair into a string tuple, (host, port).
164   * TODO JDK7: move to InetAddress.getHostString() to avoid DNS lookups.
165   * @param address an address
166   * @return an element for the address list
167   */
168  public static List<String> marshall(InetSocketAddress address) {
169    return tuple(address.getHostName(), address.getPort());
170  }
171
172  /**
173   * Require a specific address type on an endpoint
174   * @param required required type
175   * @param epr endpoint
176   * @throws InvalidRecordException if the type is wrong
177   */
178  public static void requireAddressType(String required, Endpoint epr) throws
179      InvalidRecordException {
180    if (!required.equals(epr.addressType)) {
181      throw new InvalidRecordException(
182          epr.toString(),
183          "Address type of " + epr.addressType
184          + " does not match required type of "
185          + required);
186    }
187  }
188
189  /**
190   * Get a single URI endpoint
191   * @param epr endpoint
192   * @return the uri of the first entry in the address list. Null if the endpoint
193   * itself is null
194   * @throws InvalidRecordException if the type is wrong, there are no addresses
195   * or the payload ill-formatted
196   */
197  public static List<String> retrieveAddressesUriType(Endpoint epr)
198      throws InvalidRecordException {
199    if (epr == null) {
200      return null;
201    }
202    requireAddressType(AddressTypes.ADDRESS_URI, epr);
203    List<List<String>> addresses = epr.addresses;
204    if (addresses.size() < 1) {
205      throw new InvalidRecordException(epr.toString(),
206          "No addresses in endpoint");
207    }
208    List<String> results = new ArrayList<String>(addresses.size());
209    for (List<String> address : addresses) {
210      if (address.size() != 1) {
211        throw new InvalidRecordException(epr.toString(),
212            "Address payload invalid: wrong element count: " +
213            address.size());
214      }
215      results.add(address.get(0));
216    }
217    return results;
218  }
219
220  /**
221   * Get the address URLs. Guranteed to return at least one address.
222   * @param epr endpoint
223   * @return the address as a URL
224   * @throws InvalidRecordException if the type is wrong, there are no addresses
225   * or the payload ill-formatted
226   * @throws MalformedURLException address can't be turned into a URL
227   */
228  public static List<URL> retrieveAddressURLs(Endpoint epr)
229      throws InvalidRecordException, MalformedURLException {
230    if (epr == null) {
231      throw new InvalidRecordException("", "Null endpoint");
232    }
233    List<String> addresses = retrieveAddressesUriType(epr);
234    List<URL> results = new ArrayList<URL>(addresses.size());
235    for (String address : addresses) {
236      results.add(new URL(address));
237    }
238    return results;
239  }
