053/**
054 * <p>This class is a metrics sink that uses
055 * {@link org.apache.hadoop.fs.FileSystem} to write the metrics logs.  Every
056 * hour a new directory will be created under the path specified by the
057 * <code>basepath</code> property. All metrics will be logged to a file in the
058 * current hour's directory in a file named &lt;hostname&gt;.log, where
059 * &lt;hostname&gt; is the name of the host on which the metrics logging
060 * process is running. The base path is set by the
061 * <code>&lt;prefix&gt;.sink.&lt;instance&gt;.basepath</code> property.  The
062 * time zone used to create the current hour's directory name is GMT.  If the
063 * <code>basepath</code> property isn't specified, it will default to
064 * &quot;/tmp&quot;, which is the temp directory on whatever default file
065 * system is configured for the cluster.</p>
066 *
067 * <p>The <code>&lt;prefix&gt;.sink.&lt;instance&gt;.ignore-error</code>
068 * property controls whether an exception is thrown when an error is encountered
069 * writing a log file.  The default value is <code>true</code>.  When set to
070 * <code>false</code>, file errors are quietly swallowed.</p>
071 *
072 * <p>The primary use of this class is for logging to HDFS.  As it uses
073 * {@link org.apache.hadoop.fs.FileSystem} to access the target file system,
074 * however, it can be used to write to the local file system, Amazon S3, or any
075 * other supported file system.  The base path for the sink will determine the
076 * file system used.  An unqualified path will write to the default file system
077 * set by the configuration.</p>
078 *
079 * <p>Not all file systems support the ability to append to files.  In file
080 * systems without the ability to append to files, only one writer can write to
081 * a file at a time.  To allow for concurrent writes from multiple daemons on a
082 * single host, the <code>source</code> property should be set to the name of
083 * the source daemon, e.g. <i>namenode</i>.  The value of the
084 * <code>source</code> property should typically be the same as the property's
085 * prefix.  If this property is not set, the source is taken to be
086 * <i>unknown</i>.</p>
087 *
088 * <p>Instead of appending to an existing file, by default the sink
089 * will create a new file with a suffix of &quot;.&lt;n&gt;&quet;, where
090 * <i>n</i> is the next lowest integer that isn't already used in a file name,
091 * similar to the Hadoop daemon logs.  NOTE: the file with the <b>highest</b>
092 * sequence number is the <b>newest</b> file, unlike the Hadoop daemon logs.</p>
093 *
094 * <p>For file systems that allow append, the sink supports appending to the
095 * existing file instead. If the <code>allow-append</code> property is set to
096 * true, the sink will instead append to the existing file on file systems that
097 * support appends. By default, the <code>allow-append</code> property is
098 * false.</p>
099 *
100 * <p>Note that when writing to HDFS with <code>allow-append</code> set to true,
101 * there is a minimum acceptable number of data nodes.  If the number of data
102 * nodes drops below that minimum, the append will succeed, but reading the
103 * data will fail with an IOException in the DataStreamer class.  The minimum
104 * number of data nodes required for a successful append is generally 2 or
105 * 3.</p>
106 *
107 * <p>Note also that when writing to HDFS, the file size information is not
108 * updated until the file is closed (e.g. at the top of the hour) even though
109 * the data is being written successfully. This is a known HDFS limitation that
110 * exists because of the performance cost of updating the metadata.  See
111 * <a href="https://issues.apache.org/jira/browse/HDFS-5478">HDFS-5478</a>.</p>
112 *
113 * <p>When using this sink in a secure (Kerberos) environment, two additional
114 * properties must be set: <code>keytab-key</code> and
115 * <code>principal-key</code>. <code>keytab-key</code> should contain the key by
116 * which the keytab file can be found in the configuration, for example,
117 * <code>yarn.nodemanager.keytab</code>. <code>principal-key</code> should
118 * contain the key by which the principal can be found in the configuration,
119 * for example, <code>yarn.nodemanager.principal</code>.
120 */
121@InterfaceAudience.Public
122@InterfaceStability.Evolving
123public class RollingFileSystemSink implements MetricsSink, Closeable {
124  private static final String BASEPATH_KEY = "basepath";
125  private static final String SOURCE_KEY = "source";
126  private static final String IGNORE_ERROR_KEY = "ignore-error";
127  private static final String ALLOW_APPEND_KEY = "allow-append";
128  private static final String KEYTAB_PROPERTY_KEY = "keytab-key";
129  private static final String USERNAME_PROPERTY_KEY = "principal-key";
130  private static final String SOURCE_DEFAULT = "unknown";
131  private static final String BASEPATH_DEFAULT = "/tmp";
132  private static final FastDateFormat DATE_FORMAT =
133      FastDateFormat.getInstance("yyyyMMddHH", TimeZone.getTimeZone("GMT"));
134  private final Object lock = new Object();
135  private String source;
136  private boolean ignoreError;
137  private boolean allowAppend;
138  private Path basePath;
139  private FileSystem fileSystem;
140  // The current directory path into which we're writing files
141  private Path currentDirPath;
142  // The path to the current file into which we're writing data
143  private Path currentFilePath;
144  // The stream to which we're currently writing.
145  private PrintStream currentOutStream;
146  // We keep this only to be able to call hsynch() on it.
147  private FSDataOutputStream currentFSOutStream;
148  private Timer flushTimer;

150  // This flag is used during testing to make the flusher thread run after only
151  // a short pause instead of waiting for the top of the hour.
152  @VisibleForTesting
153  protected static boolean flushQuickly = false;
154  // This flag is used by the flusher thread to indicate that it has run. Used
155  // only for testing purposes.
156  @VisibleForTesting
157  protected static volatile boolean hasFlushed = false;
158  // Use this configuration instead of loading a new one.
159  @VisibleForTesting
160  protected static Configuration suppliedConf = null;
161  // Use this file system instead of getting a new one.
162  @VisibleForTesting
163  protected static FileSystem suppliedFilesystem = null;

165  @Override
166  public void init(SubsetConfiguration conf) {
167    basePath = new Path(conf.getString(BASEPATH_KEY, BASEPATH_DEFAULT));
168    source = conf.getString(SOURCE_KEY, SOURCE_DEFAULT);
169    ignoreError = conf.getBoolean(IGNORE_ERROR_KEY, false);
170    allowAppend = conf.getBoolean(ALLOW_APPEND_KEY, false);
171
172    Configuration configuration = loadConf();
173
174    UserGroupInformation.setConfiguration(configuration);
175
176    // Don't do secure setup if it's not needed.
177    if (UserGroupInformation.isSecurityEnabled()) {
178      // Validate config so that we don't get an NPE
179      checkForProperty(conf, KEYTAB_PROPERTY_KEY);
180      checkForProperty(conf, USERNAME_PROPERTY_KEY);
181
182
183      try {
184        // Login as whoever we're supposed to be and let the hostname be pulled
185        // from localhost. If security isn't enabled, this does nothing.
186        SecurityUtil.login(configuration, conf.getString(KEYTAB_PROPERTY_KEY),
187            conf.getString(USERNAME_PROPERTY_KEY));
188      } catch (IOException ex) {
189        throw new MetricsException("Error logging in securely: ["
190            + ex.toString() + "]", ex);
191      }
192    }
193
194    fileSystem = getFileSystem(configuration);
195
196    // This step isn't strictly necessary, but it makes debugging issues much
197    // easier. We try to create the base directory eagerly and fail with
198    // copious debug info if it fails.
199    try {
200      fileSystem.mkdirs(basePath);
201    } catch (Exception ex) {
202      throw new MetricsException("Failed to create " + basePath + "["
203          + SOURCE_KEY + "=" + source + ", "
204          + IGNORE_ERROR_KEY + "=" + ignoreError + ", "
205          + ALLOW_APPEND_KEY + "=" + allowAppend + ", "
206          + KEYTAB_PROPERTY_KEY + "="
207          + conf.getString(KEYTAB_PROPERTY_KEY) + ", "
208          + conf.getString(KEYTAB_PROPERTY_KEY) + "="
209          + configuration.get(conf.getString(KEYTAB_PROPERTY_KEY)) + ", "
210          + USERNAME_PROPERTY_KEY + "="
211          + conf.getString(USERNAME_PROPERTY_KEY) + ", "
212          + conf.getString(USERNAME_PROPERTY_KEY) + "="
213          + configuration.get(conf.getString(USERNAME_PROPERTY_KEY))
214          + "] -- " + ex.toString(), ex);
215    }
216
217    // If we're permitted to append, check if we actually can
218    if (allowAppend) {
219      allowAppend = checkAppend(fileSystem);
220    }
221
222    flushTimer = new Timer("RollingFileSystemSink Flusher", true);
223  }

225  /**
226   * Throw a {@link MetricsException} if the given property is not set.
227   *
228   * @param conf the configuration to test
229   * @param key the key to validate
230   */
231  private static void checkForProperty(SubsetConfiguration conf, String key) {
232    if (!conf.containsKey(key)) {
233      throw new MetricsException("Configuration is missing " + key
234          + " property");
235    }
236  }

238  /**
239   * Return the supplied configuration for testing or otherwise load a new
240   * configuration.
241   *
242   * @return the configuration to use
243   */
244  private Configuration loadConf() {
245    Configuration conf;
246
247    if (suppliedConf != null) {
248      conf = suppliedConf;
249    } else {
250      // The config we're handed in init() isn't the one we want here, so we
251      // create a new one to pick up the full settings.
252      conf = new Configuration();
253    }
254
255    return conf;
256  }
257
258  /**
259   * Return the supplied file system for testing or otherwise get a new file
260   * system.
261   *
262   * @param conf the configuration
263   * @return the file system to use
264   * @throws MetricsException thrown if the file system could not be retrieved
265   */
266  private FileSystem getFileSystem(Configuration conf) throws MetricsException {
267    FileSystem fs = null;
268
269    if (suppliedFilesystem != null) {
270      fs = suppliedFilesystem;
271    } else {
272      try {
273        fs = FileSystem.get(new URI(basePath.toString()), conf);
274      } catch (URISyntaxException ex) {
275        throw new MetricsException("The supplied filesystem base path URI"
276            + " is not a valid URI: " + basePath.toString(), ex);
277      } catch (IOException ex) {
278        throw new MetricsException("Error connecting to file system: "
279            + basePath + " [" + ex.toString() + "]", ex);
280      }
281    }
282
283    return fs;
284  }
285
286  /**
287   * Test whether the file system supports append and return the answer.
288   * @param fs the target file system
289   */
290  private boolean checkAppend(FileSystem fs) {
291    boolean canAppend = true;
292
293    try {
294      fs.append(basePath);
295    } catch (IOException ex) {
296      if (ex.getMessage().equals("Not supported")) {
297        canAppend = false;
298      }
299    }
300
301    return canAppend;
302  }

304  /**
305   * Check the current directory against the time stamp.  If they're not
306   * the same, create a new directory and a new log file in that directory.
307   *
308   * @throws MetricsException thrown if an error occurs while creating the
309   * new directory or new log file
310   */
311  private void rollLogDirIfNeeded() throws MetricsException {
312    Date now = new Date();
313    String currentDir = DATE_FORMAT.format(now);
314    Path path = new Path(basePath, currentDir);
315
316    // We check whether currentOutStream is null instead of currentDirPath,
317    // because if currentDirPath is null, then currentOutStream is null, but
318    // currentOutStream can be null for other reasons.
319    if ((currentOutStream == null) || !path.equals(currentDirPath)) {
320      // Close the stream. This step could have been handled already by the
321      // flusher thread, but if it has, the PrintStream will just swallow the
322      // exception, which is fine.
323      if (currentOutStream != null) {
324        currentOutStream.close();
325      }
326
327      currentDirPath = path;
328
329      try {
330        rollLogDir();
331      } catch (IOException ex) {
332        throwMetricsException("Failed to create new log file", ex);
333      }
334
335      scheduleFlush(now);
336    }
337  }
338
339  /**
340   * Schedule the current hour's directory to be flushed at the top of the next
341   * hour. If this ends up running after the top of the next hour, it will
342   * execute immediately.
343   *
344   * @param now the current time
345   */
346  private void scheduleFlush(Date now) {
347    // Store the current currentDirPath to close later
348    final PrintStream toClose = currentOutStream;
349    Calendar next = Calendar.getInstance();
350
351    next.setTime(now);
352
353    if (flushQuickly) {
354      // If we're running unit tests, flush after a short pause
355      next.add(Calendar.MILLISECOND, 400);
356    } else {
357      // Otherwise flush at the top of the hour
358      next.set(Calendar.SECOND, 0);
359      next.set(Calendar.MINUTE, 0);
360      next.add(Calendar.HOUR, 1);
361    }
362
363    flushTimer.schedule(new TimerTask() {
364      @Override
365      public void run() {
366        synchronized (lock) {
367          // This close may have already been done by a putMetrics() call. If it
368          // has, the PrintStream will swallow the exception, which is fine.
369          toClose.close();
370        }
371
372        hasFlushed = true;
373      }
374    }, next.getTime());
375  }

377  /**
378   * Create a new directory based on the current hour and a new log file in
379   * that directory.
380   *
381   * @throws IOException thrown if an error occurs while creating the
382   * new directory or new log file
383   */
384  private void rollLogDir() throws IOException {
385    String fileName =
386        source + "-" + InetAddress.getLocalHost().getHostName() + ".log";
387
388    Path targetFile = new Path(currentDirPath, fileName);
389    fileSystem.mkdirs(currentDirPath);
390
391    if (allowAppend) {
392      createOrAppendLogFile(targetFile);
393    } else {
394      createLogFile(targetFile);
395    }
396  }

398  /**
399   * Create a new log file and return the {@link FSDataOutputStream}. If a
400   * file with the specified path already exists, add a suffix, starting with 1
401   * and try again. Keep incrementing the suffix until a nonexistent target
402   * path is found.
403   *
404   * Once the file is open, update {@link #currentFSOutStream},
405   * {@link #currentOutStream}, and {@#link #currentFile} are set appropriately.
406   *
407   * @param initial the target path
408   * @throws IOException thrown if the call to see if the exists fails
409   */
410  private void createLogFile(Path initial) throws IOException {
411    Path currentAttempt = initial;
412    // Start at 0 so that if the base filname exists, we start with the suffix
413    // ".1".
414    int id = 0;
415
416    while (true) {
417      // First try blindly creating the file. If we fail, it either means
418      // the file exists, or the operation actually failed.  We do it this way
419      // because if we check whether the file exists, it might still be created
420      // by the time we try to create it. Creating first works like a
421      // test-and-set.
422      try {
423        currentFSOutStream = fileSystem.create(currentAttempt, false);
424        currentOutStream = new PrintStream(currentFSOutStream, true,
425            StandardCharsets.UTF_8.name());
426        currentFilePath = currentAttempt;
427        break;
428      } catch (IOException ex) {
429        // Now we can check to see if the file exists to know why we failed
430        if (fileSystem.exists(currentAttempt)) {
431          id = getNextIdToTry(initial, id);
432          currentAttempt = new Path(initial.toString() + "." + id);
433        } else {
434          throw ex;
435        }
436      }
437    }
438  }
439
440  /**
441   * Return the next ID suffix to use when creating the log file. This method
442   * will look at the files in the directory, find the one with the highest
443   * ID suffix, and 1 to that suffix, and return it. This approach saves a full
444   * linear probe, which matters in the case where there are a large number of
445   * log files.
446   *
447   * @param initial the base file path
448   * @param lastId the last ID value that was used
449   * @return the next ID to try
450   * @throws IOException thrown if there's an issue querying the files in the
451   * directory
452   */
453  private int getNextIdToTry(Path initial, int lastId)
454      throws IOException {
455    RemoteIterator<LocatedFileStatus> files =
456        fileSystem.listFiles(currentDirPath, true);
457    String base = initial.toString();
458    int id = lastId;
459
460    while (files.hasNext()) {
461      String file = files.next().getPath().getName();
462
463      if (file.startsWith(base)) {
464        int fileId = extractId(file);
465
466        if (fileId > id) {
467          id = fileId;
468        }
469      }
470    }
471
472    // Return either 1 more than the highest we found or 1 more than the last
473    // ID used (if no ID was found).
474    return id + 1;
475  }
476
477  /**
478   * Extract the ID from the suffix of the given file name.
479   *
480   * @param file the file name
481   * @return the ID or -1 if no ID could be extracted
482   */
483  private int extractId(String file) {
484    int index = file.lastIndexOf(".");
485    int id = -1;
486
487    // A hostname has to have at least 1 character
488    if (index > 0) {
489      try {
490        id = Integer.parseInt(file.substring(index + 1));
491      } catch (NumberFormatException ex) {
492        // This can happen if there's no suffix, but there is a dot in the
493        // hostname.  Just ignore it.
494      }
495    }
496
497    return id;
498  }

500  /**
501   * Create a new log file and return the {@link FSDataOutputStream}. If a
502   * file with the specified path already exists, open the file for append
503   * instead.
504   *
505   * Once the file is open, update {@link #currentFSOutStream},
506   * {@link #currentOutStream}, and {@#link #currentFile} are set appropriately.
507   *
508   * @param initial the target path
509   * @throws IOException thrown if the call to see the append operation fails.
510   */
511  private void createOrAppendLogFile(Path targetFile) throws IOException {
512    // First try blindly creating the file. If we fail, it either means
513    // the file exists, or the operation actually failed.  We do it this way
514    // because if we check whether the file exists, it might still be created
515    // by the time we try to create it. Creating first works like a
516    // test-and-set.
517    try {
518      currentFSOutStream = fileSystem.create(targetFile, false);
519      currentOutStream = new PrintStream(currentFSOutStream, true,
520          StandardCharsets.UTF_8.name());
521    } catch (IOException ex) {
522      // Try appending instead.  If we fail, if means the file doesn't
523      // actually exist yet or the operation actually failed.
524      try {
525        currentFSOutStream = fileSystem.append(targetFile);
526        currentOutStream = new PrintStream(currentFSOutStream, true,
527            StandardCharsets.UTF_8.name());
528      } catch (IOException ex2) {
529        // If the original create failed for a legit but transitory
530        // reason, the append will fail because the file now doesn't exist,
531        // resulting in a confusing stack trace.  To avoid that, we set
532        // the cause of the second exception to be the first exception.
533        // It's still a tiny bit confusing, but it's enough
534        // information that someone should be able to figure it out.
535        ex2.initCause(ex);
536
537        throw ex2;
538      }
539    }
540
541    currentFilePath = targetFile;
542  }

544  @Override
545  public void putMetrics(MetricsRecord record) {
546    synchronized (lock) {
547      rollLogDirIfNeeded();
548
549      if (currentOutStream != null) {
550        currentOutStream.printf("%d %s.%s", record.timestamp(),
551            record.context(), record.name());
552
553        String separator = ": ";
554
555        for (MetricsTag tag : record.tags()) {
556          currentOutStream.printf("%s%s=%s", separator, tag.name(),
557              tag.value());
558          separator = ", ";
559        }
560
561        for (AbstractMetric metric : record.metrics()) {
562          currentOutStream.printf("%s%s=%s", separator, metric.name(),
563              metric.value());
564        }
565
566        currentOutStream.println();
567
568        // If we don't hflush(), the data may not be written until the file is
569        // closed. The file won't be closed until the top of the hour *AND*
570        // another record is received. Calling hflush() makes sure that the data
571        // is complete at the top of the hour.
572        try {
573          currentFSOutStream.hflush();
574        } catch (IOException ex) {
575          throwMetricsException("Failed flushing the stream", ex);
576        }
577
578        checkForErrors("Unable to write to log file");
579      } else if (!ignoreError) {
580        throwMetricsException("Unable to write to log file");
581      }
582    }
583  }
584
585  @Override
586  public void flush() {
587    synchronized (lock) {
588      // currentOutStream is null if currentFSOutStream is null
589      if (currentFSOutStream != null) {
590        try {
591          currentFSOutStream.hflush();
592        } catch (IOException ex) {
593          throwMetricsException("Unable to flush log file", ex);
594        }
595      }
596    }
597  }
598
599  @Override
600  public void close() {
601    synchronized (lock) {
602      if (currentOutStream != null) {
603        currentOutStream.close();
604
605        try {
606          checkForErrors("Unable to close log file");
607        } finally {
608          // Null out the streams just in case someone tries to reuse us.
609          currentOutStream = null;
610          currentFSOutStream = null;
611        }
612      }
613    }
614  }

616  /**
617   * If the sink isn't set to ignore errors, throw a {@link MetricsException}
618   * if the stream encountered an exception.  The message parameter will be used
619   * as the new exception's message with the current file name
620   * ({@link #currentFilePath}) appended to it.
621   *
622   * @param message the exception message. The message will have the current
623   * file name ({@link #currentFilePath}) appended to it.
624   * @throws MetricsException thrown if there was an error and the sink isn't
625   * ignoring errors
626   */
627  private void checkForErrors(String message)
628      throws MetricsException {
629    if (!ignoreError && currentOutStream.checkError()) {
630      throw new MetricsException(message + ": " + currentFilePath);
631    }
632  }

634  /**
635   * If the sink isn't set to ignore errors, wrap the Throwable in a
636   * {@link MetricsException} and throw it.  The message parameter will be used
637   * as the new exception's message with the current file name
638   * ({@link #currentFilePath}) and the Throwable's string representation
639   * appended to it.
640   *
641   * @param message the exception message. The message will have the current
642   * file name ({@link #currentFilePath}) and the Throwable's string
643   * representation appended to it.
644   * @param t the Throwable to wrap
645   */
646  private void throwMetricsException(String message, Throwable t) {
647    if (!ignoreError) {
648      throw new MetricsException(message + ": " + currentFilePath + " ["
649          + t.toString() + "]", t);
650    }
651  }
652
653  /**
654   * If the sink isn't set to ignore errors, throw a new
655   * {@link MetricsException}.  The message parameter will be used  as the
656   * new exception's message with the current file name
657   * ({@link #currentFilePath}) appended to it.
658   *
659   * @param message the exception message. The message will have the current
660   * file name ({@link #currentFilePath}) appended to it.
661   */
662  private void throwMetricsException(String message) {
663    if (!ignoreError) {
664      throw new MetricsException(message + ": " + currentFilePath);
665    }
666  }
