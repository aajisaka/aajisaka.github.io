020
021import java.io.IOException;
022import java.util.EnumSet;

024import java.util.Set;
025
026import org.apache.hadoop.classification.InterfaceAudience;

029import org.apache.hadoop.classification.InterfaceStability;

051@InterfaceAudience.Public
052@InterfaceStability.Stable
053public abstract class YarnClient extends AbstractService {
054
055  /**
056   * Create a new instance of YarnClient.
057   */
058  @Public
059  public static YarnClient createYarnClient() {
060    YarnClient client = new YarnClientImpl();
061    return client;
062  }
063
064  @Private
065  protected YarnClient(String name) {
066    super(name);
067  }
068
069  /**
070   * <p>
071   * Obtain a {@link YarnClientApplication} for a new application,
072   * which in turn contains the {@link ApplicationSubmissionContext} and
073   * {@link org.apache.hadoop.yarn.api.protocolrecords.GetNewApplicationResponse}
074   * objects.
075   * </p>
076   *
077   * @return {@link YarnClientApplication} built for a new application
078   * @throws YarnException
079   * @throws IOException
080   */
081  public abstract YarnClientApplication createApplication()
082      throws YarnException, IOException;
083
084  /**
085   * <p>
086   * Submit a new application to <code>YARN.</code> It is a blocking call, such
087   * that it will not return {@link ApplicationId} until the submitted
088   * application has been submitted and accepted by the ResourceManager.
089   * </p>
090   * 
091   * @param appContext
092   *          {@link ApplicationSubmissionContext} containing all the details
093   *          needed to submit a new application
094   * @return {@link ApplicationId} of the accepted application
095   * @throws YarnException
096   * @throws IOException
097   * @see #createApplication()
098   */
099  public abstract ApplicationId submitApplication(ApplicationSubmissionContext appContext)
100      throws YarnException, IOException;
101
102  /**
103   * <p>
104   * Kill an application identified by given ID.
105   * </p>
106   * 
107   * @param applicationId
108   *          {@link ApplicationId} of the application that needs to be killed
109   * @throws YarnException
110   *           in case of errors or if YARN rejects the request due to
111   *           access-control restrictions.
112   * @throws IOException
113   * @see #getQueueAclsInfo()
114   */
115  public abstract void killApplication(ApplicationId applicationId) throws YarnException,
116      IOException;
117
118  /**
119   * <p>
120   * Get a report of the given Application.
121   * </p>
122   * 
123   * <p>
124   * In secure mode, <code>YARN</code> verifies access to the application, queue
125   * etc. before accepting the request.
126   * </p>
127   * 
128   * <p>
129   * If the user does not have <code>VIEW_APP</code> access then the following
130   * fields in the report will be set to stubbed values:
131   * <ul>
132   * <li>host - set to "N/A"</li>
133   * <li>RPC port - set to -1</li>
134   * <li>client token - set to "N/A"</li>
135   * <li>diagnostics - set to "N/A"</li>
136   * <li>tracking URL - set to "N/A"</li>
137   * <li>original tracking URL - set to "N/A"</li>
138   * <li>resource usage report - all values are -1</li>
139   * </ul>
140   * </p>
141   * 
142   * @param appId
143   *          {@link ApplicationId} of the application that needs a report
144   * @return application report
145   * @throws YarnException
146   * @throws IOException
147   */
148  public abstract ApplicationReport getApplicationReport(ApplicationId appId)
149      throws YarnException, IOException;
150
151  /**
152   * Get the AMRM token of the application.
153   * <p/>
154   * The AMRM token is required for AM to RM scheduling operations. For 
155   * managed Application Masters Yarn takes care of injecting it. For unmanaged
156   * Applications Masters, the token must be obtained via this method and set
157   * in the {@link org.apache.hadoop.security.UserGroupInformation} of the
158   * current user.
159   * <p/>
160   * The AMRM token will be returned only if all the following conditions are
161   * met:
162   * <li>
163   *   <ul>the requester is the owner of the ApplicationMaster</ul>
164   *   <ul>the application master is an unmanaged ApplicationMaster</ul>
165   *   <ul>the application master is in ACCEPTED state</ul>
166   * </li>
167   * Else this method returns NULL.
168   *
169   * @param appId {@link ApplicationId} of the application to get the AMRM token
170   * @return the AMRM token if available
171   * @throws YarnException
172   * @throws IOException
173   */
174  public abstract org.apache.hadoop.security.token.Token<AMRMTokenIdentifier>
175      getAMRMToken(ApplicationId appId) throws YarnException, IOException;
176
177  /**
178   * <p>
179   * Get a report (ApplicationReport) of all Applications in the cluster.
180   * </p>
181   *
182   * <p>
183   * If the user does not have <code>VIEW_APP</code> access for an application
184   * then the corresponding report will be filtered as described in
185   * {@link #getApplicationReport(ApplicationId)}.
186   * </p>
187   *
188   * @return a list of reports of all running applications
189   * @throws YarnException
190   * @throws IOException
191   */
192  public abstract List<ApplicationReport> getApplications()
193      throws YarnException, IOException;
194
195  /**
196   * <p>
197   * Get a report (ApplicationReport) of Applications
198   * matching the given application types in the cluster.
199   * </p>
200   *
201   * <p>
202   * If the user does not have <code>VIEW_APP</code> access for an application
203   * then the corresponding report will be filtered as described in
204   * {@link #getApplicationReport(ApplicationId)}.
205   * </p>
206   *
207   * @param applicationTypes
208   * @return a list of reports of applications
209   * @throws YarnException
210   * @throws IOException
211   */
212  public abstract List<ApplicationReport> getApplications(
213      Set<String> applicationTypes) throws YarnException, IOException;
214
215  /**
216   * <p>
217   * Get a report (ApplicationReport) of Applications matching the given
218   * application states in the cluster.
219   * </p>
220   *
221   * <p>
222   * If the user does not have <code>VIEW_APP</code> access for an application
223   * then the corresponding report will be filtered as described in
224   * {@link #getApplicationReport(ApplicationId)}.
225   * </p>
226   *
227   * @param applicationStates
228   * @return a list of reports of applications
229   * @throws YarnException
230   * @throws IOException
231   */
232  public abstract List<ApplicationReport>
233      getApplications(EnumSet<YarnApplicationState> applicationStates)
234          throws YarnException, IOException;
235
236  /**
237   * <p>
238   * Get a report (ApplicationReport) of Applications matching the given
239   * application types and application states in the cluster.
240   * </p>
241   *
242   * <p>
243   * If the user does not have <code>VIEW_APP</code> access for an application
244   * then the corresponding report will be filtered as described in
245   * {@link #getApplicationReport(ApplicationId)}.
246   * </p>
247   *
248   * @param applicationTypes
249   * @param applicationStates
250   * @return a list of reports of applications
251   * @throws YarnException
252   * @throws IOException
253   */
254  public abstract List<ApplicationReport> getApplications(
255      Set<String> applicationTypes,
256      EnumSet<YarnApplicationState> applicationStates) throws YarnException,
257      IOException;
258
259  /**
260   * <p>
261   * Get metrics ({@link YarnClusterMetrics}) about the cluster.
262   * </p>
263   * 
264   * @return cluster metrics
265   * @throws YarnException
266   * @throws IOException
267   */
268  public abstract YarnClusterMetrics getYarnClusterMetrics() throws YarnException,
269      IOException;
270
271  /**
272   * <p>
273   * Get a report of nodes ({@link NodeReport}) in the cluster.
274   * </p>
275   * 
276   * @param states The {@link NodeState}s to filter on. If no filter states are
277   *          given, nodes in all states will be returned.
278   * @return A list of node reports
279   * @throws YarnException
280   * @throws IOException
281   */
282  public abstract List<NodeReport> getNodeReports(NodeState... states)
283      throws YarnException, IOException;
284
285  /**
286   * <p>
287   * Get a delegation token so as to be able to talk to YARN using those tokens.
288   * 
289   * @param renewer
290   *          Address of the renewer who can renew these tokens when needed by
291   *          securely talking to YARN.
292   * @return a delegation token ({@link Token}) that can be used to
293   *         talk to YARN
294   * @throws YarnException
295   * @throws IOException
296   */
297  public abstract Token getRMDelegationToken(Text renewer)
298      throws YarnException, IOException;
299
300  /**
301   * <p>
302   * Get information ({@link QueueInfo}) about a given <em>queue</em>.
303   * </p>
304   * 
305   * @param queueName
306   *          Name of the queue whose information is needed
307   * @return queue information
308   * @throws YarnException
309   *           in case of errors or if YARN rejects the request due to
310   *           access-control restrictions.
311   * @throws IOException
312   */
313  public abstract QueueInfo getQueueInfo(String queueName) throws YarnException,
314      IOException;
315
316  /**
317   * <p>
318   * Get information ({@link QueueInfo}) about all queues, recursively if there
319   * is a hierarchy
320   * </p>
321   * 
322   * @return a list of queue-information for all queues
323   * @throws YarnException
324   * @throws IOException
325   */
326  public abstract List<QueueInfo> getAllQueues() throws YarnException, IOException;
327
328  /**
329   * <p>
330   * Get information ({@link QueueInfo}) about top level queues.
331   * </p>
332   * 
333   * @return a list of queue-information for all the top-level queues
334   * @throws YarnException
335   * @throws IOException
336   */
337  public abstract List<QueueInfo> getRootQueueInfos() throws YarnException, IOException;
338
339  /**
340   * <p>
341   * Get information ({@link QueueInfo}) about all the immediate children queues
342   * of the given queue
343   * </p>
344   * 
345   * @param parent
346   *          Name of the queue whose child-queues' information is needed
347   * @return a list of queue-information for all queues who are direct children
348   *         of the given parent queue.
349   * @throws YarnException
350   * @throws IOException
351   */
352  public abstract List<QueueInfo> getChildQueueInfos(String parent) throws YarnException,
353      IOException;
354
355  /**
356   * <p>
357   * Get information about <em>acls</em> for <em>current user</em> on all the
358   * existing queues.
359   * </p>
360   * 
361   * @return a list of queue acls ({@link QueueUserACLInfo}) for
362   *         <em>current user</em>
363   * @throws YarnException
364   * @throws IOException
365   */
366  public abstract List<QueueUserACLInfo> getQueueAclsInfo() throws YarnException,
367      IOException;
368  
369  /**
370   * <p>
371   * Get a report of the given ApplicationAttempt.
372   * </p>
373   * 
374   * <p>
375   * In secure mode, <code>YARN</code> verifies access to the application, queue
376   * etc. before accepting the request.
377   * </p>
378   * 
379   * @param applicationAttemptId
380   *          {@link ApplicationAttemptId} of the application attempt that needs
381   *          a report
382   * @return application attempt report
383   * @throws YarnException
384   * @throws {@link ApplicationAttemptNotFoundException} if application attempt
385   *         not found
386   * @throws IOException
387   */
388  public abstract ApplicationAttemptReport getApplicationAttemptReport(
389      ApplicationAttemptId applicationAttemptId) throws YarnException, IOException;
390
391  /**
392   * <p>
393   * Get a report of all (ApplicationAttempts) of Application in the cluster.
394   * </p>
395   * 
396   * @param applicationId
397   * @return a list of reports for all application attempts for specified
398   *         application.
399   * @throws YarnException
400   * @throws IOException
401   */
402  public abstract List<ApplicationAttemptReport> getApplicationAttempts(
403      ApplicationId applicationId) throws YarnException, IOException;
404
405  /**
406   * <p>
407   * Get a report of the given Container.
408   * </p>
409   * 
410   * <p>
411   * In secure mode, <code>YARN</code> verifies access to the application, queue
412   * etc. before accepting the request.
413   * </p>
414   * 
415   * @param containerId
416   *          {@link ContainerId} of the container that needs a report
417   * @return container report
418   * @throws YarnException
419   * @throws {@link ContainerNotFoundException} if container not found.
420   * @throws IOException
421   */
422  public abstract ContainerReport getContainerReport(ContainerId containerId)
423      throws YarnException, IOException;
424
425  /**
426   * <p>
427   * Get a report of all (Containers) of ApplicationAttempt in the cluster.
428   * </p>
429   * 
430   * @param applicationAttemptId
431   * @return a list of reports of all containers for specified application
432   *         attempts
433   * @throws YarnException
434   * @throws IOException
435   */
436  public abstract List<ContainerReport> getContainers(
437      ApplicationAttemptId applicationAttemptId) throws YarnException,
438      IOException;
439  
440  /**
441   * <p>
442   * Attempts to move the given application to the given queue.
443   * </p>
444   * 
445   * @param appId
446   *    Application to move.
447   * @param queue
448   *    Queue to place it in to.
449   * @throws YarnException
450   * @throws IOException
451   */
452  public abstract void moveApplicationAcrossQueues(ApplicationId appId,
453      String queue) throws YarnException, IOException;
