1277  /**
1278   * Get the socket address for <code>name</code> property as a
1279   * <code>InetSocketAddress</code>. On a HA cluster,
1280   * this fetches the address corresponding to the RM identified by
1281   * {@link #RM_HA_ID}.
1282   * @param name property name.
1283   * @param defaultAddress the default value
1284   * @param defaultPort the default port
1285   * @return InetSocketAddress
1286   */
1287  @Override
1288  public InetSocketAddress getSocketAddr(
1289      String name, String defaultAddress, int defaultPort) {
1290    String address;
1291    if (HAUtil.isHAEnabled(this) && getServiceAddressConfKeys(this).contains(name)) {
1292      address = HAUtil.getConfValueForRMInstance(name, defaultAddress, this);
1293    } else {
1294      address = get(name, defaultAddress);
1295    }
1296    return NetUtils.createSocketAddr(address, defaultPort, name);
1297  }

1301                                             InetSocketAddress addr) {

1303    if (HAUtil.isHAEnabled(this)) {
1304      prefix = HAUtil.addSuffix(prefix, HAUtil.getRMHAId(this));
1305    }
1306    return super.updateConnectAddr(prefix, addr);

1309  @Private
1310  public static int getRMDefaultPortNumber(String addressPrefix,
1311      Configuration conf) {
1312    if (addressPrefix.equals(YarnConfiguration.RM_ADDRESS)) {
1313      return YarnConfiguration.DEFAULT_RM_PORT;
1314    } else if (addressPrefix.equals(YarnConfiguration.RM_SCHEDULER_ADDRESS)) {
1315      return YarnConfiguration.DEFAULT_RM_SCHEDULER_PORT;
1316    } else if (addressPrefix.equals(YarnConfiguration.RM_WEBAPP_ADDRESS)) {
1317      return YarnConfiguration.DEFAULT_RM_WEBAPP_PORT;
1318    } else if (addressPrefix.equals(YarnConfiguration.RM_WEBAPP_HTTPS_ADDRESS)) {
1319      return YarnConfiguration.DEFAULT_RM_WEBAPP_HTTPS_PORT;
1320    } else if (addressPrefix
1321        .equals(YarnConfiguration.RM_RESOURCE_TRACKER_ADDRESS)) {
1322      return YarnConfiguration.DEFAULT_RM_RESOURCE_TRACKER_PORT;
1323    } else if (addressPrefix.equals(YarnConfiguration.RM_ADMIN_ADDRESS)) {
1324      return YarnConfiguration.DEFAULT_RM_ADMIN_PORT;
1325    } else {
1326      throw new HadoopIllegalArgumentException(
1327          "Invalid RM RPC address Prefix: " + addressPrefix
1328              + ". The valid value should be one of "
1329              + getServiceAddressConfKeys(conf));
1330    }
1331  }
