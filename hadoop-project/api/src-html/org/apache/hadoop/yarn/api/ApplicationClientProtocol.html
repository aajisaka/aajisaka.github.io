063/**
064 * <p>The protocol between clients and the <code>ResourceManager</code>
065 * to submit/abort jobs and to get information on applications, cluster metrics,
066 * nodes, queues and ACLs.</p> 
067 */
068@Public
069@Stable
070public interface ApplicationClientProtocol {
071  /**
072   * <p>The interface used by clients to obtain a new {@link ApplicationId} for 
073   * submitting new applications.</p>
074   * 
075   * <p>The <code>ResourceManager</code> responds with a new, monotonically
076   * increasing, {@link ApplicationId} which is used by the client to submit
077   * a new application.</p>
078   *
079   * <p>The <code>ResourceManager</code> also responds with details such 
080   * as maximum resource capabilities in the cluster as specified in
081   * {@link GetNewApplicationResponse}.</p>
082   *
083   * @param request request to get a new <code>ApplicationId</code>
084   * @return response containing the new <code>ApplicationId</code> to be used
085   * to submit an application
086   * @throws YarnException
087   * @throws IOException
088   * @see #submitApplication(SubmitApplicationRequest)
089   */
090  @Public
091  @Stable
092  public GetNewApplicationResponse getNewApplication(
093      GetNewApplicationRequest request)
094  throws YarnException, IOException;
095  
096  /**
097   * <p>The interface used by clients to submit a new application to the
098   * <code>ResourceManager.</code></p>
099   * 
100   * <p>The client is required to provide details such as queue, 
101   * {@link Resource} required to run the <code>ApplicationMaster</code>, 
102   * the equivalent of {@link ContainerLaunchContext} for launching
103   * the <code>ApplicationMaster</code> etc. via the 
104   * {@link SubmitApplicationRequest}.</p>
105   * 
106   * <p>Currently the <code>ResourceManager</code> sends an immediate (empty) 
107   * {@link SubmitApplicationResponse} on accepting the submission and throws 
108   * an exception if it rejects the submission. However, this call needs to be
109   * followed by {@link #getApplicationReport(GetApplicationReportRequest)}
110   * to make sure that the application gets properly submitted.</p>
111   * 
112   * <p> In secure mode,the <code>ResourceManager</code> verifies access to
113   * queues etc. before accepting the application submission.</p>
114   * 
115   * @param request request to submit a new application
116   * @return (empty) response on accepting the submission
117   * @throws YarnException
118   * @throws IOException
119   * @throws InvalidResourceRequestException
120   *           The exception is thrown when a {@link ResourceRequest} is out of
121   *           the range of the configured lower and upper resource boundaries.
122   * @see #getNewApplication(GetNewApplicationRequest)
123   */
124  @Public
125  @Stable
126  public SubmitApplicationResponse submitApplication(
127      SubmitApplicationRequest request) 
128  throws YarnException, IOException;
129  
130  /**
131   * <p>The interface used by clients to request the 
132   * <code>ResourceManager</code> to abort submitted application.</p>
133   * 
134   * <p>The client, via {@link KillApplicationRequest} provides the
135   * {@link ApplicationId} of the application to be aborted.</p>
136   * 
137   * <p> In secure mode,the <code>ResourceManager</code> verifies access to the
138   * application, queue etc. before terminating the application.</p> 
139   * 
140   * <p>Currently, the <code>ResourceManager</code> returns an empty response
141   * on success and throws an exception on rejecting the request.</p>
142   * 
143   * @param request request to abort a submitted application
144   * @return <code>ResourceManager</code> returns an empty response
145   *         on success and throws an exception on rejecting the request
146   * @throws YarnException
147   * @throws IOException
148   * @see #getQueueUserAcls(GetQueueUserAclsInfoRequest) 
149   */
150  @Public
151  @Stable
152  public KillApplicationResponse forceKillApplication(
153      KillApplicationRequest request) 
154  throws YarnException, IOException;
155
156  /**
157   * <p>The interface used by clients to get a report of an Application from
158   * the <code>ResourceManager</code>.</p>
159   * 
160   * <p>The client, via {@link GetApplicationReportRequest} provides the
161   * {@link ApplicationId} of the application.</p>
162   *
163   * <p> In secure mode,the <code>ResourceManager</code> verifies access to the
164   * application, queue etc. before accepting the request.</p> 
165   * 
166   * <p>The <code>ResourceManager</code> responds with a 
167   * {@link GetApplicationReportResponse} which includes the 
168   * {@link ApplicationReport} for the application.</p>
169   * 
170   * <p>If the user does not have <code>VIEW_APP</code> access then the
171   * following fields in the report will be set to stubbed values:
172   * <ul>
173   *   <li>host - set to "N/A"</li>
174   *   <li>RPC port - set to -1</li>
175   *   <li>client token - set to "N/A"</li>
176   *   <li>diagnostics - set to "N/A"</li>
177   *   <li>tracking URL - set to "N/A"</li>
178   *   <li>original tracking URL - set to "N/A"</li>
179   *   <li>resource usage report - all values are -1</li>
180   * </ul></p>
181   *
182   * @param request request for an application report
183   * @return application report 
184   * @throws YarnException
185   * @throws IOException
186   */
187  @Public
188  @Stable
189  public GetApplicationReportResponse getApplicationReport(
190      GetApplicationReportRequest request) 
191  throws YarnException, IOException;
192  
193  /**
194   * <p>The interface used by clients to get metrics about the cluster from
195   * the <code>ResourceManager</code>.</p>
196   * 
197   * <p>The <code>ResourceManager</code> responds with a
198   * {@link GetClusterMetricsResponse} which includes the 
199   * {@link YarnClusterMetrics} with details such as number of current
200   * nodes in the cluster.</p>
201   * 
202   * @param request request for cluster metrics
203   * @return cluster metrics
204   * @throws YarnException
205   * @throws IOException
206   */
207  @Public
208  @Stable
209  public GetClusterMetricsResponse getClusterMetrics(
210      GetClusterMetricsRequest request) 
211  throws YarnException, IOException;
212  
213  /**
214   * <p>The interface used by clients to get a report of Applications
215   * matching the filters defined by {@link GetApplicationsRequest}
216   * in the cluster from the <code>ResourceManager</code>.</p>
217   * 
218   * <p>The <code>ResourceManager</code> responds with a 
219   * {@link GetApplicationsResponse} which includes the
220   * {@link ApplicationReport} for the applications.</p>
221   * 
222   * <p>If the user does not have <code>VIEW_APP</code> access for an
223   * application then the corresponding report will be filtered as
224   * described in {@link #getApplicationReport(GetApplicationReportRequest)}.
225   * </p>
226   *
227   * @param request request for report on applications
228   * @return report on applications matching the given application types
229   *           defined in the request
230   * @throws YarnException
231   * @throws IOException
232   * @see GetApplicationsRequest
233   */
234  @Public
235  @Stable
236  public GetApplicationsResponse getApplications(
237      GetApplicationsRequest request)
238  throws YarnException, IOException;
239  
240  /**
241   * <p>The interface used by clients to get a report of all nodes
242   * in the cluster from the <code>ResourceManager</code>.</p>
243   * 
244   * <p>The <code>ResourceManager</code> responds with a 
245   * {@link GetClusterNodesResponse} which includes the 
246   * {@link NodeReport} for all the nodes in the cluster.</p>
247   * 
248   * @param request request for report on all nodes
249   * @return report on all nodes
250   * @throws YarnException
251   * @throws IOException
252   */
253  @Public
254  @Stable
255  public GetClusterNodesResponse getClusterNodes(
256      GetClusterNodesRequest request) 
257  throws YarnException, IOException;
258  
259  /**
260   * <p>The interface used by clients to get information about <em>queues</em>
261   * from the <code>ResourceManager</code>.</p>
262   * 
263   * <p>The client, via {@link GetQueueInfoRequest}, can ask for details such
264   * as used/total resources, child queues, running applications etc.</p>
265   *
266   * <p> In secure mode,the <code>ResourceManager</code> verifies access before
267   * providing the information.</p> 
268   * 
269   * @param request request to get queue information
270   * @return queue information
271   * @throws YarnException
272   * @throws IOException
273   */
274  @Public
275  @Stable
276  public GetQueueInfoResponse getQueueInfo(
277      GetQueueInfoRequest request) 
278  throws YarnException, IOException;
279  
280  /**
281   * <p>The interface used by clients to get information about <em>queue 
282   * acls</em> for <em>current user</em> from the <code>ResourceManager</code>.
283   * </p>
284   * 
285   * <p>The <code>ResourceManager</code> responds with queue acls for all
286   * existing queues.</p>
287   * 
288   * @param request request to get queue acls for <em>current user</em>
289   * @return queue acls for <em>current user</em>
290   * @throws YarnException
291   * @throws IOException
292   */
293  @Public
294  @Stable
295  public GetQueueUserAclsInfoResponse getQueueUserAcls(
296      GetQueueUserAclsInfoRequest request) 
297  throws YarnException, IOException;
298  
299  /**
300   * <p>The interface used by clients to get delegation token, enabling the 
301   * containers to be able to talk to the service using those tokens.
302   * 
303   *  <p> The <code>ResourceManager</code> responds with the delegation
304   *  {@link Token} that can be used by the client to speak to this
305   *  service.
306   * @param request request to get a delegation token for the client.
307   * @return delegation token that can be used to talk to this service
308   * @throws YarnException
309   * @throws IOException
310   */
311  @Public
312  @Stable
313  public GetDelegationTokenResponse getDelegationToken(
314      GetDelegationTokenRequest request) 
315  throws YarnException, IOException;
316  
317  /**
318   * Renew an existing delegation {@link Token}.
319   * 
320   * @param request the delegation token to be renewed.
321   * @return the new expiry time for the delegation token.
322   * @throws YarnException
323   * @throws IOException
324   */
325  @Private
326  @Unstable
327  public RenewDelegationTokenResponse renewDelegationToken(
328      RenewDelegationTokenRequest request) throws YarnException,
329      IOException;
330
331  /**
332   * Cancel an existing delegation {@link Token}.
333   * 
334   * @param request the delegation token to be cancelled.
335   * @return an empty response.
336   * @throws YarnException
337   * @throws IOException
338   */
339  @Private
340  @Unstable
341  public CancelDelegationTokenResponse cancelDelegationToken(
342      CancelDelegationTokenRequest request) throws YarnException,
343      IOException;
344  
345  /**
346   * Move an application to a new queue.
347   * 
348   * @param request the application ID and the target queue
349   * @return an empty response
350   * @throws YarnException
351   * @throws IOException
352   */
353  @Public
354  @Unstable
355  public MoveApplicationAcrossQueuesResponse moveApplicationAcrossQueues(
356      MoveApplicationAcrossQueuesRequest request) throws YarnException, IOException;
357}
