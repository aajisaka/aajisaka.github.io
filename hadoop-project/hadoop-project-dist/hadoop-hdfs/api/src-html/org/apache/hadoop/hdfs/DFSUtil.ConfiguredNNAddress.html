155  /** Shuffle the elements in the given array. */
156  public static <T> T[] shuffle(final T[] array) {
157    if (array != null && array.length > 0) {
158      final Random random = getRandom();
159      for (int n = array.length; n > 1; ) {
160        final int randomIndex = random.nextInt(n);
161        n--;
162        if (n != randomIndex) {
163          final T tmp = array[randomIndex];
164          array[randomIndex] = array[n];
165          array[n] = tmp;
166        }
167      }
168    }
169    return array;
170  }

187  /**
188   * Comparator for sorting DataNodeInfo[] based on decommissioned/stale states.
189   * Decommissioned/stale nodes are moved to the end of the array on sorting
190   * with this comparator.
191   */ 
192  @InterfaceAudience.Private 
193  public static class DecomStaleComparator implements Comparator<DatanodeInfo> {
194    private final long staleInterval;
195
196    /**
197     * Constructor of DecomStaleComparator
198     * 
199     * @param interval
200     *          The time interval for marking datanodes as stale is passed from
201     *          outside, since the interval may be changed dynamically
202     */
203    public DecomStaleComparator(long interval) {
204      this.staleInterval = interval;
205    }
206
207    @Override
208    public int compare(DatanodeInfo a, DatanodeInfo b) {
209      // Decommissioned nodes will still be moved to the end of the list
210      if (a.isDecommissioned()) {
211        return b.isDecommissioned() ? 0 : 1;
212      } else if (b.isDecommissioned()) {
213        return -1;
214      }
215      // Stale nodes will be moved behind the normal nodes
216      boolean aStale = a.isStale(staleInterval);
217      boolean bStale = b.isStale(staleInterval);
218      return aStale == bStale ? 0 : (aStale ? 1 : -1);
219    }
220  }

232  /**
233   * Whether the pathname is valid.  Currently prohibits relative paths, 
234   * names which contain a ":" or "//", or other non-canonical paths.
235   */
236  public static boolean isValidName(String src) {
237    // Path must be absolute.
238    if (!src.startsWith(Path.SEPARATOR)) {
239      return false;
240    }
241      
242    // Check for ".." "." ":" "/"
243    String[] components = StringUtils.split(src, '/');
244    for (int i = 0; i < components.length; i++) {
245      String element = components[i];
246      if (element.equals(".")  ||
247          (element.indexOf(":") >= 0)  ||
248          (element.indexOf("/") >= 0)) {
249        return false;
250      }
251      // ".." is allowed in path starting with /.reserved/.inodes
252      if (element.equals("..")) {
253        if (components.length > 4
254            && components[1].equals(FSDirectory.DOT_RESERVED_STRING)
255            && components[2].equals(FSDirectory.DOT_INODES_STRING)) {
256          continue;
257        }
258        return false;
259      }
260      // The string may start or end with a /, but not have
261      // "//" in the middle.
262      if (element.isEmpty() && i != components.length - 1 &&
263          i != 0) {
264        return false;
265      }
266    }
267    return true;
268  }

341  /**
342   * Given a list of path components returns a path as a UTF8 String
343   */
344  public static String byteArray2PathString(byte[][] pathComponents,
345      int offset, int length) {
346    if (pathComponents.length == 0) {
347      return "";
348    }
349    Preconditions.checkArgument(offset >= 0 && offset < pathComponents.length);
350    Preconditions.checkArgument(length >= 0 && offset + length <=
351        pathComponents.length);
352    if (pathComponents.length == 1
353        && (pathComponents[0] == null || pathComponents[0].length == 0)) {
354      return Path.SEPARATOR;
355    }
356    StringBuilder result = new StringBuilder();
357    for (int i = offset; i < offset + length; i++) {
358      result.append(new String(pathComponents[i], Charsets.UTF_8));
359      if (i < pathComponents.length - 1) {
360        result.append(Path.SEPARATOR_CHAR);
361      }
362    }
363    return result.toString();
364  }

388  /**
389   * Given a list of path components returns a byte array
390   */
391  public static byte[] byteArray2bytes(byte[][] pathComponents) {
392    if (pathComponents.length == 0) {
393      return EMPTY_BYTES;
394    } else if (pathComponents.length == 1
395        && (pathComponents[0] == null || pathComponents[0].length == 0)) {
396      return new byte[]{(byte) Path.SEPARATOR_CHAR};
397    }
398    int length = 0;
399    for (int i = 0; i < pathComponents.length; i++) {
400      length += pathComponents[i].length;
401      if (i < pathComponents.length - 1) {
402        length++; // for SEPARATOR
403      }
404    }
405    byte[] path = new byte[length];
406    int index = 0;
407    for (int i = 0; i < pathComponents.length; i++) {
408      System.arraycopy(pathComponents[i], 0, path, index,
409          pathComponents[i].length);
410      index += pathComponents[i].length;
411      if (i < pathComponents.length - 1) {
412        path[index] = (byte) Path.SEPARATOR_CHAR;
413        index++;
414      }
415    }
416    return path;
417  }

437  /**
438   * Splits first len bytes in bytes to array of arrays of bytes
439   * on byte separator
440   * @param bytes the byte array to split
441   * @param len the number of bytes to split
442   * @param separator the delimiting byte
443   */
444  public static byte[][] bytes2byteArray(byte[] bytes,
445                                         int len,
446                                         byte separator) {
447    assert len <= bytes.length;
448    int splits = 0;
449    if (len == 0) {
450      return new byte[][]{null};
451    }
452    // Count the splits. Omit multiple separators and the last one
453    for (int i = 0; i < len; i++) {
454      if (bytes[i] == separator) {
455        splits++;
456      }
457    }
458    int last = len - 1;
459    while (last > -1 && bytes[last--] == separator) {
460      splits--;
461    }
462    if (splits == 0 && bytes[0] == separator) {
463      return new byte[][]{null};
464    }
465    splits++;
466    byte[][] result = new byte[splits][];
467    int startIndex = 0;
468    int nextIndex = 0;
469    int index = 0;
470    // Build the splits
471    while (index < splits) {
472      while (nextIndex < len && bytes[nextIndex] != separator) {
473        nextIndex++;
474      }
475      result[index] = new byte[nextIndex - startIndex];
476      System.arraycopy(bytes, startIndex, result[index], 0, nextIndex
477              - startIndex);
478      index++;
479      startIndex = nextIndex + 1;
480      nextIndex = startIndex;
481    }
482    return result;
483  }

497  /**
498   * Convert a List<LocatedBlock> to BlockLocation[]
499   * @param blocks A List<LocatedBlock> to be converted
500   * @return converted array of BlockLocation
501   */
502  public static BlockLocation[] locatedBlocks2Locations(List<LocatedBlock> blocks) {
503    if (blocks == null) {
504      return new BlockLocation[0];
505    }
506    int nrBlocks = blocks.size();
507    BlockLocation[] blkLocations = new BlockLocation[nrBlocks];
508    if (nrBlocks == 0) {
509      return blkLocations;
510    }
511    int idx = 0;
512    for (LocatedBlock blk : blocks) {
513      assert idx < nrBlocks : "Incorrect index";
514      DatanodeInfo[] locations = blk.getLocations();
515      String[] hosts = new String[locations.length];
516      String[] xferAddrs = new String[locations.length];
517      String[] racks = new String[locations.length];
518      for (int hCnt = 0; hCnt < locations.length; hCnt++) {
519        hosts[hCnt] = locations[hCnt].getHostName();
520        xferAddrs[hCnt] = locations[hCnt].getXferAddr();
521        NodeBase node = new NodeBase(xferAddrs[hCnt], 
522                                     locations[hCnt].getNetworkLocation());
523        racks[hCnt] = node.toString();
524      }
525      DatanodeInfo[] cachedLocations = blk.getCachedLocations();
526      String[] cachedHosts = new String[cachedLocations.length];
527      for (int i=0; i<cachedLocations.length; i++) {
528        cachedHosts[i] = cachedLocations[i].getHostName();
529      }
530      blkLocations[idx] = new BlockLocation(xferAddrs, hosts, cachedHosts,
531                                            racks,
532                                            blk.getStartOffset(),
533                                            blk.getBlockSize(),
534                                            blk.isCorrupt());
535      idx++;
536    }
537    return blkLocations;
538  }

982  /**
983   * Get a URI for each configured nameservice. If a nameservice is
984   * HA-enabled, then the logical URI of the nameservice is returned. If the
985   * nameservice is not HA-enabled, then a URI corresponding to the address of
986   * the single NN for that nameservice is returned.
987   * 
988   * @param conf configuration
989   * @param keys configuration keys to try in order to get the URI for non-HA
990   *        nameservices
991   * @return a collection of all configured NN URIs
992   */
993  public static Collection<URI> getNameServiceUris(Configuration conf,
994      String... keys) {
995    Set<URI> ret = new HashSet<URI>();
996    
997    // We're passed multiple possible configuration keys for any given NN or HA
998    // nameservice, and search the config in order of these keys. In order to
999    // make sure that a later config lookup (e.g. fs.defaultFS) doesn't add a
1000    // URI for a config key for which we've already found a preferred entry, we
1001    // keep track of non-preferred keys here.
1002    Set<URI> nonPreferredUris = new HashSet<URI>();
1003    
1004    for (String nsId : getNameServiceIds(conf)) {
1005      if (HAUtil.isHAEnabled(conf, nsId)) {
1006        // Add the logical URI of the nameservice.
1007        try {
1008          ret.add(new URI(HdfsConstants.HDFS_URI_SCHEME + "://" + nsId));
1009        } catch (URISyntaxException ue) {
1010          throw new IllegalArgumentException(ue);
1011        }
1012      } else {
1013        // Add the URI corresponding to the address of the NN.
1014        boolean uriFound = false;
1015        for (String key : keys) {
1016          String addr = conf.get(concatSuffixes(key, nsId));
1017          if (addr != null) {
1018            URI uri = createUri(HdfsConstants.HDFS_URI_SCHEME,
1019                NetUtils.createSocketAddr(addr));
1020            if (!uriFound) {
1021              uriFound = true;
1022              ret.add(uri);
1023            } else {
1024              nonPreferredUris.add(uri);
1025            }
1026          }
1027        }
1028      }
1029    }
1030    
1031    // Add the generic configuration keys.
1032    boolean uriFound = false;
1033    for (String key : keys) {
1034      String addr = conf.get(key);
1035      if (addr != null) {
1036        URI uri = createUri("hdfs", NetUtils.createSocketAddr(addr));
1037        if (!uriFound) {
1038          uriFound = true;
1039          ret.add(uri);
1040        } else {
1041          nonPreferredUris.add(uri);
1042        }
1043      }
1044    }
1045    
1046    // Add the default URI if it is an HDFS URI.
1047    URI defaultUri = FileSystem.getDefaultUri(conf);
1048    // checks if defaultUri is ip:port format
1049    // and convert it to hostname:port format
1050    if (defaultUri != null && (defaultUri.getPort() != -1)) {
1051      defaultUri = createUri(defaultUri.getScheme(),
1052          NetUtils.createSocketAddr(defaultUri.getHost(), 
1053              defaultUri.getPort()));
1054    }
1055    if (defaultUri != null &&
1056        HdfsConstants.HDFS_URI_SCHEME.equals(defaultUri.getScheme()) &&
1057        !nonPreferredUris.contains(defaultUri)) {
1058      ret.add(defaultUri);
1059    }
1060    
1061    return ret;
1062  }

1093   * return server http or https address from the configuration for a
1094   * given namenode rpc address.
1095   * @param namenodeAddr - namenode RPC address
1096   * @param conf configuration
1097   * @param scheme - the scheme (http / https)
1098   * @return server http or https address
1099   * @throws IOException 
1100   */
1101  public static URI getInfoServer(InetSocketAddress namenodeAddr,
1102      Configuration conf, String scheme) throws IOException {
1103    String[] suffixes = null;
1104    if (namenodeAddr != null) {
1105      // if non-default namenode, try reverse look up 
1106      // the nameServiceID if it is available
1107      suffixes = getSuffixIDs(conf, namenodeAddr,
1108          DFSConfigKeys.DFS_NAMENODE_SERVICE_RPC_ADDRESS_KEY,
1109          DFSConfigKeys.DFS_NAMENODE_RPC_ADDRESS_KEY);
1110    }
1111
1112    String authority;
1113    if ("http".equals(scheme)) {
1114      authority = getSuffixedConf(conf, DFS_NAMENODE_HTTP_ADDRESS_KEY,
1115          DFS_NAMENODE_HTTP_ADDRESS_DEFAULT, suffixes);
1116    } else if ("https".equals(scheme)) {
1117      authority = getSuffixedConf(conf, DFS_NAMENODE_HTTPS_ADDRESS_KEY,
1118          DFS_NAMENODE_HTTPS_ADDRESS_DEFAULT, suffixes);
1119    } else {
1120      throw new IllegalArgumentException("Invalid scheme:" + scheme);
1121    }
1122
1123    if (namenodeAddr != null) {
1124      authority = substituteForWildcardAddress(authority,
1125          namenodeAddr.getHostName());
1126    }
1127    return URI.create(scheme + "://" + authority);
1128  }

1165   * with a wildcard. This is used, for example, when determining the HTTP
1166   * address of the NN -- if it's configured to bind to 0.0.0.0, we want to
1167   * substitute the hostname from the filesystem URI rather than trying to
1168   * connect to 0.0.0.0.

1170   * @param defaultHost the host to substitute with, if configuredAddress

1172   * @return the substituted address
1173   * @throws IOException if it is a wildcard address and security is enabled

1175  @VisibleForTesting
1176  static String substituteForWildcardAddress(String configuredAddress,
1177    String defaultHost) throws IOException {
1178    InetSocketAddress sockAddr = NetUtils.createSocketAddr(configuredAddress);
1179    InetSocketAddress defaultSockAddr = NetUtils.createSocketAddr(defaultHost
1180        + ":0");
1181    final InetAddress addr = sockAddr.getAddress();
1182    if (addr != null && addr.isAnyLocalAddress()) {
1183      if (UserGroupInformation.isSecurityEnabled() &&
1184          defaultSockAddr.getAddress().isAnyLocalAddress()) {
1185        throw new IOException("Cannot use a wildcard address with security. " +
1186            "Must explicitly set bind address for Kerberos");
1187      }
1188      return defaultHost + ":" + sockAddr.getPort();
1189    } else {
1190      return configuredAddress;
1191    }

1338  /**
1339   * Returns nameservice Id and namenode Id when the local host matches the
1340   * configuration parameter {@code addressKey}.<nameservice Id>.<namenode Id>
1341   * 
1342   * @param conf Configuration
1343   * @param addressKey configuration key corresponding to the address.
1344   * @param knownNsId only look at configs for the given nameservice, if not-null
1345   * @param knownNNId only look at configs for the given namenode, if not null
1346   * @param matcher matching criteria for matching the address
1347   * @return Array with nameservice Id and namenode Id on success. First element
1348   *         in the array is nameservice Id and second element is namenode Id.
1349   *         Null value indicates that the configuration does not have the the
1350   *         Id.
1351   * @throws HadoopIllegalArgumentException on error
1352   */
1353  static String[] getSuffixIDs(final Configuration conf, final String addressKey,
1354      String knownNsId, String knownNNId,
1355      final AddressMatcher matcher) {
1356    String nameserviceId = null;
1357    String namenodeId = null;
1358    int found = 0;
1359    
1360    Collection<String> nsIds = getNameServiceIds(conf);
1361    for (String nsId : emptyAsSingletonNull(nsIds)) {
1362      if (knownNsId != null && !knownNsId.equals(nsId)) {
1363        continue;
1364      }
1365      
1366      Collection<String> nnIds = getNameNodeIds(conf, nsId);
1367      for (String nnId : emptyAsSingletonNull(nnIds)) {
1368        if (LOG.isTraceEnabled()) {
1369          LOG.trace(String.format("addressKey: %s nsId: %s nnId: %s",
1370              addressKey, nsId, nnId));
1371        }
1372        if (knownNNId != null && !knownNNId.equals(nnId)) {
1373          continue;
1374        }
1375        String key = addKeySuffixes(addressKey, nsId, nnId);
1376        String addr = conf.get(key);
1377        if (addr == null) {
1378          continue;
1379        }
1380        InetSocketAddress s = null;
1381        try {
1382          s = NetUtils.createSocketAddr(addr);
1383        } catch (Exception e) {
1384          LOG.warn("Exception in creating socket address " + addr, e);
1385          continue;
1386        }
1387        if (!s.isUnresolved() && matcher.match(s)) {
1388          nameserviceId = nsId;
1389          namenodeId = nnId;
1390          found++;
1391        }
1392      }
1393    }
1394    if (found > 1) { // Only one address must match the local address
1395      String msg = "Configuration has multiple addresses that match "
1396          + "local node's address. Please configure the system with "
1397          + DFS_NAMESERVICE_ID + " and "
1398          + DFS_HA_NAMENODE_ID_KEY;
1399      throw new HadoopIllegalArgumentException(msg);
1400    }
1401    return new String[] { nameserviceId, namenodeId };
1402  }

1404  /**
1405   * For given set of {@code keys} adds nameservice Id and or namenode Id
1406   * and returns {nameserviceId, namenodeId} when address match is found.
1407   * @see #getSuffixIDs(Configuration, String, String, String, AddressMatcher)
1408   */
1409  static String[] getSuffixIDs(final Configuration conf,
1410      final InetSocketAddress address, final String... keys) {
1411    AddressMatcher matcher = new AddressMatcher() {
1412     @Override
1413      public boolean match(InetSocketAddress s) {
1414        return address.equals(s);
1415      } 
1416    };
1417    
1418    for (String key : keys) {
1419      String[] ids = getSuffixIDs(conf, key, null, null, matcher);
1420      if (ids != null && (ids [0] != null || ids[1] != null)) {
1421        return ids;
1422      }
1423    }
1424    return null;
1425  }

1653  /**
1654   * Converts a time duration in milliseconds into DDD:HH:MM:SS format.
1655   */
1656  public static String durationToString(long durationMs) {
1657    boolean negative = false;
1658    if (durationMs < 0) {
1659      negative = true;
1660      durationMs = -durationMs;
1661    }
1662    // Chop off the milliseconds
1663    long durationSec = durationMs / 1000;
1664    final int secondsPerMinute = 60;
1665    final int secondsPerHour = 60*60;
1666    final int secondsPerDay = 60*60*24;
1667    final long days = durationSec / secondsPerDay;
1668    durationSec -= days * secondsPerDay;
1669    final long hours = durationSec / secondsPerHour;
1670    durationSec -= hours * secondsPerHour;
1671    final long minutes = durationSec / secondsPerMinute;
1672    durationSec -= minutes * secondsPerMinute;
1673    final long seconds = durationSec;
1674    final long milliseconds = durationMs % 1000;
1675    String format = "%03d:%02d:%02d:%02d.%03d";
1676    if (negative)  {
1677      format = "-" + format;
1678    }
1679    return String.format(format, days, hours, minutes, seconds, milliseconds);
1680  }

1682  /**
1683   * Converts a relative time string into a duration in milliseconds.
1684   */
1685  public static long parseRelativeTime(String relTime) throws IOException {
1686    if (relTime.length() < 2) {
1687      throw new IOException("Unable to parse relative time value of " + relTime
1688          + ": too short");
1689    }
1690    String ttlString = relTime.substring(0, relTime.length()-1);
1691    long ttl;
1692    try {
1693      ttl = Long.parseLong(ttlString);
1694    } catch (NumberFormatException e) {
1695      throw new IOException("Unable to parse relative time value of " + relTime
1696          + ": " + ttlString + " is not a number");
1697    }
1698    if (relTime.endsWith("s")) {
1699      // pass
1700    } else if (relTime.endsWith("m")) {
1701      ttl *= 60;
1702    } else if (relTime.endsWith("h")) {
1703      ttl *= 60*60;
1704    } else if (relTime.endsWith("d")) {
1705      ttl *= 60*60*24;
1706    } else {
1707      throw new IOException("Unable to parse relative time value of " + relTime
1708          + ": unknown time unit " + relTime.charAt(relTime.length() - 1));
1709    }
1710    return ttl*1000;
1711  }
