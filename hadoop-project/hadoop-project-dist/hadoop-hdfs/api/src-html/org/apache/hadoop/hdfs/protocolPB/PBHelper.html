309  public static BlockStoragePolicy convert(BlockStoragePolicyProto proto) {
310    List<StorageTypeProto> cList = proto.getCreationPolicy()
311        .getStorageTypesList();
312    StorageType[] creationTypes = convertStorageTypes(cList, cList.size());
313    List<StorageTypeProto> cfList = proto.hasCreationFallbackPolicy() ? proto
314        .getCreationFallbackPolicy().getStorageTypesList() : null;
315    StorageType[] creationFallbackTypes = cfList == null ? StorageType
316        .EMPTY_ARRAY : convertStorageTypes(cfList, cfList.size());
317    List<StorageTypeProto> rfList = proto.hasReplicationFallbackPolicy() ?
318        proto.getReplicationFallbackPolicy().getStorageTypesList() : null;
319    StorageType[] replicationFallbackTypes = rfList == null ? StorageType
320        .EMPTY_ARRAY : convertStorageTypes(rfList, rfList.size());
321    return new BlockStoragePolicy((byte) proto.getPolicyId(), proto.getName(),
322        creationTypes, creationFallbackTypes, replicationFallbackTypes);
323  }

325  public static BlockStoragePolicyProto convert(BlockStoragePolicy policy) {
326    BlockStoragePolicyProto.Builder builder = BlockStoragePolicyProto
327        .newBuilder().setPolicyId(policy.getId()).setName(policy.getName());
328    // creation storage types
329    StorageTypesProto creationProto = convert(policy.getStorageTypes());
330    Preconditions.checkArgument(creationProto != null);
331    builder.setCreationPolicy(creationProto);
332    // creation fallback
333    StorageTypesProto creationFallbackProto = convert(
334        policy.getCreationFallbacks());
335    if (creationFallbackProto != null) {
336      builder.setCreationFallbackPolicy(creationFallbackProto);
337    }
338    // replication fallback
339    StorageTypesProto replicationFallbackProto = convert(
340        policy.getReplicationFallbacks());
341    if (replicationFallbackProto != null) {
342      builder.setReplicationFallbackPolicy(replicationFallbackProto);
343    }
344    return builder.build();
345  }

764  public static LocatedBlockProto convert(LocatedBlock b) {
765    if (b == null) return null;
766    Builder builder = LocatedBlockProto.newBuilder();
767    DatanodeInfo[] locs = b.getLocations();
768    List<DatanodeInfo> cachedLocs =
769        Lists.newLinkedList(Arrays.asList(b.getCachedLocations()));
770    for (int i = 0; i < locs.length; i++) {
771      DatanodeInfo loc = locs[i];
772      builder.addLocs(i, PBHelper.convert(loc));
773      boolean locIsCached = cachedLocs.contains(loc);
774      builder.addIsCached(locIsCached);
775      if (locIsCached) {
776        cachedLocs.remove(loc);
777      }
778    }
779    Preconditions.checkArgument(cachedLocs.size() == 0,
780        "Found additional cached replica locations that are not in the set of"
781        + " storage-backed locations!");
782
783    StorageType[] storageTypes = b.getStorageTypes();
784    if (storageTypes != null) {
785      for (int i = 0; i < storageTypes.length; ++i) {
786        builder.addStorageTypes(PBHelper.convertStorageType(storageTypes[i]));
787      }
788    }
789    final String[] storageIDs = b.getStorageIDs();
790    if (storageIDs != null) {
791      builder.addAllStorageIDs(Arrays.asList(storageIDs));
792    }
793
794    return builder.setB(PBHelper.convert(b.getBlock()))
795        .setBlockToken(PBHelper.convert(b.getBlockToken()))
796        .setCorrupt(b.isCorrupt()).setOffset(b.getStartOffset()).build();
797  }

799  public static LocatedBlock convert(LocatedBlockProto proto) {
800    if (proto == null) return null;
801    List<DatanodeInfoProto> locs = proto.getLocsList();
802    DatanodeInfo[] targets = new DatanodeInfo[locs.size()];
803    for (int i = 0; i < locs.size(); i++) {
804      targets[i] = PBHelper.convert(locs.get(i));
805    }
806
807    final StorageType[] storageTypes = convertStorageTypes(
808        proto.getStorageTypesList(), locs.size());
809
810    final int storageIDsCount = proto.getStorageIDsCount();
811    final String[] storageIDs;
812    if (storageIDsCount == 0) {
813      storageIDs = null;
814    } else {
815      Preconditions.checkState(storageIDsCount == locs.size());
816      storageIDs = proto.getStorageIDsList().toArray(new String[storageIDsCount]);
817    }
818
819    // Set values from the isCached list, re-using references from loc
820    List<DatanodeInfo> cachedLocs = new ArrayList<DatanodeInfo>(locs.size());
821    List<Boolean> isCachedList = proto.getIsCachedList();
822    for (int i=0; i<isCachedList.size(); i++) {
823      if (isCachedList.get(i)) {
824        cachedLocs.add(targets[i]);
825      }
826    }
827
828    LocatedBlock lb = new LocatedBlock(PBHelper.convert(proto.getB()), targets,
829        storageIDs, storageTypes, proto.getOffset(), proto.getCorrupt(),
830        cachedLocs.toArray(new DatanodeInfo[0]));
831    lb.setBlockToken(PBHelper.convert(proto.getBlockToken()));
832
833    return lb;
834  }
