051  /** Startup options for rolling upgrade. */
052  public static enum RollingUpgradeStartupOption{
053    ROLLBACK, STARTED;
054
055    public String getOptionString() {
056      return StartupOption.ROLLINGUPGRADE.getName() + " "
057          + StringUtils.toLowerCase(name());
058    }
059
060    public boolean matches(StartupOption option) {
061      return option == StartupOption.ROLLINGUPGRADE
062          && option.getRollingUpgradeStartupOption() == this;
063    }
064
065    private static final RollingUpgradeStartupOption[] VALUES = values();
066
067    static RollingUpgradeStartupOption fromString(String s) {
068      if ("downgrade".equalsIgnoreCase(s)) {
069        throw new IllegalArgumentException(
070            "The \"downgrade\" option is no longer supported"
071                + " since it may incorrectly finalize an ongoing rolling upgrade."
072                + " For downgrade instruction, please see the documentation"
073                + " (http://hadoop.apache.org/docs/current/hadoop-project-dist/"
074                + "hadoop-hdfs/HdfsRollingUpgrade.html#Downgrade).");
075      }
076      for(RollingUpgradeStartupOption opt : VALUES) {
077        if (opt.name().equalsIgnoreCase(s)) {
078          return opt;
079        }
080      }
081      throw new IllegalArgumentException("Failed to convert \"" + s
082          + "\" to " + RollingUpgradeStartupOption.class.getSimpleName());
083    }
084
085    public static String getAllOptionString() {
086      final StringBuilder b = new StringBuilder("<");
087      for(RollingUpgradeStartupOption opt : VALUES) {
088        b.append(StringUtils.toLowerCase(opt.name())).append("|");
089      }
090      b.setCharAt(b.length() - 1, '>');
091      return b.toString();
092    }
093  }

098    CLUSTERID ("-clusterid"),
099    GENCLUSTERID ("-genclusterid"),

122    private static final Pattern ENUM_WITH_ROLLING_UPGRADE_OPTION = Pattern.compile(
123        "(\\w+)\\((\\w+)\\)");
124
125    private final String name;
126    
127    // Used only with format and upgrade options
128    private String clusterId = null;
129    
130    // Used only by rolling upgrade
131    private RollingUpgradeStartupOption rollingUpgradeStartupOption;
132
133    // Used only with format option
134    private boolean isForceFormat = false;
135    private boolean isInteractiveFormat = true;
136    
137    // Used only with recovery option
138    private int force = 0;
139
140    private StartupOption(String arg) {this.name = arg;}
141    public String getName() {return name;}
142    public NamenodeRole toNodeRole() {
143      switch(this) {
144      case BACKUP: 
145        return NamenodeRole.BACKUP;
146      case CHECKPOINT: 
147        return NamenodeRole.CHECKPOINT;
148      default:
149        return NamenodeRole.NAMENODE;
150      }
151    }
152    
153    public void setClusterId(String cid) {
154      clusterId = cid;
155    }
156
157    public String getClusterId() {
158      return clusterId;
159    }
160    
161    public void setRollingUpgradeStartupOption(String opt) {
162      Preconditions.checkState(this == ROLLINGUPGRADE);
163      rollingUpgradeStartupOption = RollingUpgradeStartupOption.fromString(opt);
164    }
165    
166    public RollingUpgradeStartupOption getRollingUpgradeStartupOption() {
167      Preconditions.checkState(this == ROLLINGUPGRADE);
168      return rollingUpgradeStartupOption;
169    }
170
171    public MetaRecoveryContext createRecoveryContext() {
172      if (!name.equals(RECOVER.name))
173        return null;
174      return new MetaRecoveryContext(force);
175    }
176
177    public void setForce(int force) {
178      this.force = force;
179    }
180

182      return this.force;
183    }
184    
185    public boolean getForceFormat() {
186      return isForceFormat;
187    }
188    
189    public void setForceFormat(boolean force) {
190      isForceFormat = force;
191    }
192    
193    public boolean getInteractiveFormat() {
194      return isInteractiveFormat;
195    }
196    
197    public void setInteractiveFormat(boolean interactive) {
198      isInteractiveFormat = interactive;
199    }
200    
201    @Override
202    public String toString() {
203      if (this == ROLLINGUPGRADE) {
204        return new StringBuilder(super.toString())
205            .append("(").append(getRollingUpgradeStartupOption()).append(")")
206            .toString();
207      }
208      return super.toString();
209    }
210
211    static public StartupOption getEnum(String value) {
212      Matcher matcher = ENUM_WITH_ROLLING_UPGRADE_OPTION.matcher(value);
213      if (matcher.matches()) {
214        StartupOption option = StartupOption.valueOf(matcher.group(1));
215        option.setRollingUpgradeStartupOption(matcher.group(2));
216        return option;
217      } else {
218        return StartupOption.valueOf(value);
219      }
220    }
221  }

246  /**
247   * Block replica states, which it can go through while being constructed.
248   */
249  static public enum ReplicaState {
250    /** Replica is finalized. The state when replica is not modified. */
251    FINALIZED(0),
252    /** Replica is being written to. */
253    RBW(1),
254    /** Replica is waiting to be recovered. */
255    RWR(2),
256    /** Replica is under recovery. */
257    RUR(3),
258    /** Temporary replica: created for replication and relocation only. */
259    TEMPORARY(4);
260
261    private final int value;
262
263    private ReplicaState(int v) {
264      value = v;
265    }
266
267    public int getValue() {
268      return value;
269    }
270
271    public static ReplicaState getState(int v) {
272      return ReplicaState.values()[v];
273    }
274
275    /** Read from in */
276    public static ReplicaState read(DataInput in) throws IOException {
277      return values()[in.readByte()];
278    }
279
280    /** Write to out */
281    public void write(DataOutput out) throws IOException {
282      out.writeByte(ordinal());
283    }
284  }

286  /**
287   * States, which a block can go through while it is under construction.
288   */
289  static public enum BlockUCState {
290    /**
291     * Block construction completed.<br>
292     * The block has at least the configured minimal replication number
293     * of {@link ReplicaState#FINALIZED} replica(s), and is not going to be
294     * modified.
295     * NOTE, in some special cases, a block may be forced to COMPLETE state,
296     * even if it doesn't have required minimal replications.
297     */
298    COMPLETE,
299    /**
300     * The block is under construction.<br>
301     * It has been recently allocated for write or append.
302     */
303    UNDER_CONSTRUCTION,
304    /**
305     * The block is under recovery.<br>
306     * When a file lease expires its last block may not be {@link #COMPLETE}
307     * and needs to go through a recovery procedure, 
308     * which synchronizes the existing replicas contents.
309     */
310    UNDER_RECOVERY,
311    /**
312     * The block is committed.<br>
313     * The client reported that all bytes are written to data-nodes
314     * with the given generation stamp and block length, but no 
315     * {@link ReplicaState#FINALIZED} 
316     * replicas has yet been reported by data-nodes themselves.
317     */
318    COMMITTED;
319  }
