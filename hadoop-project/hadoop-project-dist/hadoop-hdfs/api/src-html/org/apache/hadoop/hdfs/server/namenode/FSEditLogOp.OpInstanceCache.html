337  private static final class AclEditLogUtil {
338    private static final int ACL_EDITLOG_ENTRY_HAS_NAME_OFFSET = 6;
339    private static final int ACL_EDITLOG_ENTRY_TYPE_OFFSET = 3;
340    private static final int ACL_EDITLOG_ENTRY_SCOPE_OFFSET = 5;
341    private static final int ACL_EDITLOG_PERM_MASK = 7;
342    private static final int ACL_EDITLOG_ENTRY_TYPE_MASK = 3;
343    private static final int ACL_EDITLOG_ENTRY_SCOPE_MASK = 1;
344
345    private static final FsAction[] FSACTION_VALUES = FsAction.values();
346    private static final AclEntryScope[] ACL_ENTRY_SCOPE_VALUES = AclEntryScope
347        .values();
348    private static final AclEntryType[] ACL_ENTRY_TYPE_VALUES = AclEntryType
349        .values();
350
351    private static List<AclEntry> read(DataInputStream in, int logVersion)
352        throws IOException {
353      if (!NameNodeLayoutVersion.supports(Feature.EXTENDED_ACL, logVersion)) {
354        return null;
355      }
356
357      int size = in.readInt();
358      if (size == 0) {
359        return null;
360      }
361
362      List<AclEntry> aclEntries = Lists.newArrayListWithCapacity(size);
363      for (int i = 0; i < size; ++i) {
364        int v = in.read();
365        int p = v & ACL_EDITLOG_PERM_MASK;
366        int t = (v >> ACL_EDITLOG_ENTRY_TYPE_OFFSET)
367            & ACL_EDITLOG_ENTRY_TYPE_MASK;
368        int s = (v >> ACL_EDITLOG_ENTRY_SCOPE_OFFSET)
369            & ACL_EDITLOG_ENTRY_SCOPE_MASK;
370        boolean hasName = ((v >> ACL_EDITLOG_ENTRY_HAS_NAME_OFFSET) & 1) == 1;
371        String name = hasName ? FSImageSerialization.readString(in) : null;
372        aclEntries.add(new AclEntry.Builder().setName(name)
373            .setPermission(FSACTION_VALUES[p])
374            .setScope(ACL_ENTRY_SCOPE_VALUES[s])
375            .setType(ACL_ENTRY_TYPE_VALUES[t]).build());
376      }
377
378      return aclEntries;
379    }
380
381    private static void write(List<AclEntry> aclEntries, DataOutputStream out)
382        throws IOException {
383      if (aclEntries == null) {
384        out.writeInt(0);
385        return;
386      }
387
388      out.writeInt(aclEntries.size());
389      for (AclEntry e : aclEntries) {
390        boolean hasName = e.getName() != null;
391        int v = (e.getScope().ordinal() << ACL_EDITLOG_ENTRY_SCOPE_OFFSET)
392            | (e.getType().ordinal() << ACL_EDITLOG_ENTRY_TYPE_OFFSET)
393            | e.getPermission().ordinal();
394
395        if (hasName) {
396          v |= 1 << ACL_EDITLOG_ENTRY_HAS_NAME_OFFSET;
397        }
398        out.write(v);
399        if (hasName) {
400          FSImageSerialization.writeString(e.getName(), out);
401        }
402      }
403    }
404  }

569    void readFields(DataInputStream in, int logVersion)
570        throws IOException {
571      if (!NameNodeLayoutVersion.supports(
572          LayoutVersion.Feature.EDITLOG_OP_OPTIMIZATION, logVersion)) {
573        this.length = in.readInt();
574      }
575      if (NameNodeLayoutVersion.supports(
576          LayoutVersion.Feature.ADD_INODE_ID, logVersion)) {
577        this.inodeId = in.readLong();
578      } else {
579        // The inodeId should be updated when this editLogOp is applied
580        this.inodeId = INodeId.GRANDFATHER_INODE_ID;
581      }
582      if ((-17 < logVersion && length != 4) ||
583          (logVersion <= -17 && length != 5 && !NameNodeLayoutVersion.supports(
584              LayoutVersion.Feature.EDITLOG_OP_OPTIMIZATION, logVersion))) {
585        throw new IOException("Incorrect data format."  +
586                              " logVersion is " + logVersion +
587                              " but writables.length is " +
588                              length + ". ");
589      }

591
592      if (NameNodeLayoutVersion.supports(
593          LayoutVersion.Feature.EDITLOG_OP_OPTIMIZATION, logVersion)) {
594        this.replication = FSImageSerialization.readShort(in);
595        this.mtime = FSImageSerialization.readLong(in);
596      } else {
597        this.replication = readShort(in);
598        this.mtime = readLong(in);
599      }
600
601      if (NameNodeLayoutVersion.supports(
602          LayoutVersion.Feature.FILE_ACCESS_TIME, logVersion)) {
603        if (NameNodeLayoutVersion.supports(
604            LayoutVersion.Feature.EDITLOG_OP_OPTIMIZATION, logVersion)) {
605          this.atime = FSImageSerialization.readLong(in);
606        } else {
607          this.atime = readLong(in);
608        }
609      } else {
610        this.atime = 0;
611      }
612
613      if (NameNodeLayoutVersion.supports(
614          LayoutVersion.Feature.EDITLOG_OP_OPTIMIZATION, logVersion)) {
615        this.blockSize = FSImageSerialization.readLong(in);
616      } else {
617        this.blockSize = readLong(in);
618      }
619
620      this.blocks = readBlocks(in, logVersion);
621      this.permissions = PermissionStatus.read(in);
622
623      if (this.opCode == OP_ADD) {
624        aclEntries = AclEditLogUtil.read(in, logVersion);

650    private static Block[] readBlocks(
651        DataInputStream in,
652        int logVersion) throws IOException {
653      int numBlocks = in.readInt();
654      if (numBlocks < 0) {
655        throw new IOException("invalid negative number of blocks");
656      } else if (numBlocks > MAX_BLOCKS) {
657        throw new IOException("invalid number of blocks: " + numBlocks +
658            ".  The maximum number of blocks per file is " + MAX_BLOCKS);

662        Block blk = new Block();
663        blk.readFields(in);
664        blocks[i] = blk;
665      }
666      return blocks;
667    }

4609    private void verifyTerminator() throws IOException {
4610      /** The end of the edit log should contain only 0x00 or 0xff bytes.
4611       * If it contains other bytes, the log itself may be corrupt.
4612       * It is important to check this; if we don't, a stray OP_INVALID byte 
4613       * could make us stop reading the edit log halfway through, and we'd never
4614       * know that we had lost data.
4615       */
4616      byte[] buf = new byte[4096];
4617      limiter.clearLimit();
4618      int numRead = -1, idx = 0;
4619      while (true) {
4620        try {
4621          numRead = -1;
4622          idx = 0;
4623          numRead = in.read(buf);
4624          if (numRead == -1) {
4625            return;
4626          }
4627          while (idx < numRead) {
4628            if ((buf[idx] != (byte)0) && (buf[idx] != (byte)-1)) {
4629              throw new IOException("Read extra bytes after " +
4630                "the terminator!");
4631            }
4632            idx++;
4633          }
4634        } finally {
4635          // After reading each group of bytes, we reposition the mark one
4636          // byte before the next group.  Similarly, if there is an error, we
4637          // want to reposition the mark one byte before the error
4638          if (numRead != -1) { 
4639            in.reset();
4640            IOUtils.skipFully(in, idx);
4641            in.mark(buf.length + 1);
4642            IOUtils.skipFully(in, 1);
4643          }
4644        }
4645      }
4646    }

4650     *
4651     * @return   the opcode, or null on EOF.
4652     *

4654     * problematic byte.  This usually means the beginning of the opcode.
4655     */
4656    private FSEditLogOp decodeOp() throws IOException {
4657      limiter.setLimit(maxOpSize);
4658      in.mark(maxOpSize);
4659
4660      if (checksum != null) {
4661        checksum.reset();
4662      }
4663
4664      byte opCodeByte;
4665      try {
4666        opCodeByte = in.readByte();

4669        return null;
4670      }
4671

4673      if (opCode == OP_INVALID) {
4674        verifyTerminator();
4675        return null;
4676      }
4677
4678      FSEditLogOp op = cache.get(opCode);
4679      if (op == null) {
4680        throw new IOException("Read invalid opcode " + opCode);
4681      }
4682
4683      if (supportEditLogLength) {
4684        in.readInt();
4685      }
4686

4688          LayoutVersion.Feature.STORED_TXIDS, logVersion)) {
4689        // Read the txid
4690        op.setTransactionId(in.readLong());
4691      } else {
4692        op.setTransactionId(HdfsConstants.INVALID_TXID);
4693      }
4694
4695      op.readFields(in, logVersion);
4696
4697      validateChecksum(in, checksum, op.txid);
4698      return op;
4699    }

4704     * @return the last txid of the segment, or INVALID_TXID on exception
4705     */

4707      if (supportEditLogLength) {
4708        limiter.setLimit(maxOpSize);
4709        in.mark(maxOpSize);

4722        }
4723
4724        int length = in.readInt(); // read the length of the op
4725        long txid = in.readLong(); // read the txid
4726
4727        // skip the remaining content
4728        IOUtils.skipFully(in, length - 8); 
4729        // TODO: do we want to verify checksum for JN? For now we don't.
4730        return txid;
4731      } else {
4732        FSEditLogOp op = decodeOp();
4733        return op == null ? HdfsConstants.INVALID_TXID : op.getTransactionId();
4734      }
4735    }

4737    /**
4738     * Validate a transaction's checksum
4739     */
4740    private void validateChecksum(DataInputStream in,
4741                                  Checksum checksum,
4742                                  long txid)
4743        throws IOException {
4744      if (checksum != null) {
4745        int calculatedChecksum = (int)checksum.getValue();
4746        int readChecksum = in.readInt(); // read in checksum
4747        if (readChecksum != calculatedChecksum) {
4748          throw new ChecksumException(
4749              "Transaction is corrupt. Calculated checksum is " +
4750              calculatedChecksum + " but read checksum " + readChecksum, txid);
4751        }
4752      }
4753    }
4754  }
