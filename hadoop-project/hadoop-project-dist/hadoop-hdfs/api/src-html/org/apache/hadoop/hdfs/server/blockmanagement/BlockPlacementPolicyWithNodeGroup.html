032/** The class is responsible for choosing the desired number of targets
033 * for placing block replicas on environment with node-group layer.
034 * The replica placement strategy is adjusted to:
035 * If the writer is on a datanode, the 1st replica is placed on the local 
036 *     node (or local node-group), otherwise a random datanode. 
037 * The 2nd replica is placed on a datanode that is on a different rack with 1st
038 *     replica node. 
039 * The 3rd replica is placed on a datanode which is on a different node-group
040 *     but the same rack as the second replica node.
041 */
042public class BlockPlacementPolicyWithNodeGroup extends BlockPlacementPolicyDefault {
043
044  protected BlockPlacementPolicyWithNodeGroup(Configuration conf,  FSClusterStats stats,
045      NetworkTopology clusterMap, DatanodeManager datanodeManager) {
046    initialize(conf, stats, clusterMap, host2datanodeMap);
047  }
048
049  protected BlockPlacementPolicyWithNodeGroup() {
050  }

052  @Override
053  public void initialize(Configuration conf,  FSClusterStats stats,
054          NetworkTopology clusterMap, 
055          Host2NodesMap host2datanodeMap) {
056    super.initialize(conf, stats, clusterMap, host2datanodeMap);

059  /** choose local node of localMachine as the target.
060   * if localMachine is not available, choose a node on the same nodegroup or 
061   * rack instead.
062   * @return the chosen node
063   */
064  @Override
065  protected DatanodeStorageInfo chooseLocalStorage(Node localMachine,
066      Set<Node> excludedNodes, long blocksize, int maxNodesPerRack,
067      List<DatanodeStorageInfo> results, boolean avoidStaleNodes,
068      EnumMap<StorageType, Integer> storageTypes, boolean fallbackToLocalRack)
069      throws NotEnoughReplicasException {
070    // if no local machine, randomly choose one node
071    if (localMachine == null)
072      return chooseRandom(NodeBase.ROOT, excludedNodes, 
073          blocksize, maxNodesPerRack, results, avoidStaleNodes, storageTypes);
074
075    // otherwise try local machine first
076    if (localMachine instanceof DatanodeDescriptor) {
077      DatanodeDescriptor localDataNode = (DatanodeDescriptor)localMachine;
078      if (excludedNodes.add(localMachine)) { // was not in the excluded list
079        for (Iterator<Map.Entry<StorageType, Integer>> iter = storageTypes
080            .entrySet().iterator(); iter.hasNext(); ) {
081          Map.Entry<StorageType, Integer> entry = iter.next();
082          for (DatanodeStorageInfo localStorage : DFSUtil.shuffle(
083              localDataNode.getStorageInfos())) {
084            StorageType type = entry.getKey();
085            if (addIfIsGoodTarget(localStorage, excludedNodes, blocksize,
086                maxNodesPerRack, false, results, avoidStaleNodes, type) >= 0) {
087              int num = entry.getValue();
088              if (num == 1) {
089                iter.remove();
090              } else {
091                entry.setValue(num - 1);
092              }
093              return localStorage;
094            }
095          }
096        }
097      }
098    }
099
100    // try a node on local node group
101    DatanodeStorageInfo chosenStorage = chooseLocalNodeGroup(
102        (NetworkTopologyWithNodeGroup)clusterMap, localMachine, excludedNodes, 
103        blocksize, maxNodesPerRack, results, avoidStaleNodes, storageTypes);
104    if (chosenStorage != null) {
105      return chosenStorage;
106    }
107
108    if (!fallbackToLocalRack) {
109      return null;
110    }
111    // try a node on local rack
112    return chooseLocalRack(localMachine, excludedNodes, 
113        blocksize, maxNodesPerRack, results, avoidStaleNodes, storageTypes);
114  }
115
116  /** @return the node of the second replica */
117  private static DatanodeDescriptor secondNode(Node localMachine,
118      List<DatanodeStorageInfo> results) {
119    // find the second replica
120    for(DatanodeStorageInfo nextStorage : results) {
121      DatanodeDescriptor nextNode = nextStorage.getDatanodeDescriptor();
122      if (nextNode != localMachine) {
123        return nextNode;
124      }
125    }
126    return null;
127  }

129  @Override
130  protected DatanodeStorageInfo chooseLocalRack(Node localMachine,
131      Set<Node> excludedNodes, long blocksize, int maxNodesPerRack,
132      List<DatanodeStorageInfo> results, boolean avoidStaleNodes,
133      EnumMap<StorageType, Integer> storageTypes) throws
134      NotEnoughReplicasException {
135    // no local machine, so choose a random machine
136    if (localMachine == null) {
137      return chooseRandom(NodeBase.ROOT, excludedNodes, blocksize,
138          maxNodesPerRack, results, avoidStaleNodes, storageTypes);
139    }
140
141    // choose one from the local rack, but off-nodegroup
142    try {
143      final String scope = NetworkTopology.getFirstHalf(localMachine.getNetworkLocation());
144      return chooseRandom(scope, excludedNodes, blocksize, maxNodesPerRack,
145          results, avoidStaleNodes, storageTypes);
146    } catch (NotEnoughReplicasException e1) {
147      // find the second replica
148      final DatanodeDescriptor newLocal = secondNode(localMachine, results);
149      if (newLocal != null) {
150        try {
151          return chooseRandom(
152              clusterMap.getRack(newLocal.getNetworkLocation()), excludedNodes,
153              blocksize, maxNodesPerRack, results, avoidStaleNodes,
154              storageTypes);
155        } catch(NotEnoughReplicasException e2) {
156          //otherwise randomly choose one from the network
157          return chooseRandom(NodeBase.ROOT, excludedNodes, blocksize,
158              maxNodesPerRack, results, avoidStaleNodes, storageTypes);
159        }
160      } else {
161        //otherwise randomly choose one from the network
162        return chooseRandom(NodeBase.ROOT, excludedNodes, blocksize,
163            maxNodesPerRack, results, avoidStaleNodes, storageTypes);
164      }
165    }
166  }

168  @Override
169  protected void chooseRemoteRack(int numOfReplicas,
170      DatanodeDescriptor localMachine, Set<Node> excludedNodes,
171      long blocksize, int maxReplicasPerRack, List<DatanodeStorageInfo> results,
172      boolean avoidStaleNodes, EnumMap<StorageType, Integer> storageTypes)
173      throws NotEnoughReplicasException {
174    int oldNumOfReplicas = results.size();
175
176    final String rackLocation = NetworkTopology.getFirstHalf(
177        localMachine.getNetworkLocation());
178    try {
179      // randomly choose from remote racks
180      chooseRandom(numOfReplicas, "~" + rackLocation, excludedNodes, blocksize,
181          maxReplicasPerRack, results, avoidStaleNodes, storageTypes);
182    } catch (NotEnoughReplicasException e) {
183      // fall back to the local rack
184      chooseRandom(numOfReplicas - (results.size() - oldNumOfReplicas),
185          rackLocation, excludedNodes, blocksize,
186          maxReplicasPerRack, results, avoidStaleNodes, storageTypes);
187    }
188  }

190  /* choose one node from the nodegroup that <i>localMachine</i> is on.
191   * if no such node is available, choose one node from the nodegroup where
192   * a second replica is on.
193   * if still no such node is available, choose a random node in the cluster.
194   * @return the chosen node
195   */
196  private DatanodeStorageInfo chooseLocalNodeGroup(
197      NetworkTopologyWithNodeGroup clusterMap, Node localMachine,
198      Set<Node> excludedNodes, long blocksize, int maxNodesPerRack,
199      List<DatanodeStorageInfo> results, boolean avoidStaleNodes,
200      EnumMap<StorageType, Integer> storageTypes) throws
201      NotEnoughReplicasException {
202    // no local machine, so choose a random machine
203    if (localMachine == null) {
204      return chooseRandom(NodeBase.ROOT, excludedNodes, blocksize,
205          maxNodesPerRack, results, avoidStaleNodes, storageTypes);
206    }
207
208    // choose one from the local node group
209    try {
210      return chooseRandom(
211          clusterMap.getNodeGroup(localMachine.getNetworkLocation()),
212          excludedNodes, blocksize, maxNodesPerRack, results, avoidStaleNodes,
213          storageTypes);
214    } catch (NotEnoughReplicasException e1) {
215      final DatanodeDescriptor newLocal = secondNode(localMachine, results);
216      if (newLocal != null) {
217        try {
218          return chooseRandom(
219              clusterMap.getNodeGroup(newLocal.getNetworkLocation()),
220              excludedNodes, blocksize, maxNodesPerRack, results,
221              avoidStaleNodes, storageTypes);
222        } catch(NotEnoughReplicasException e2) {
223          //otherwise randomly choose one from the network
224          return chooseRandom(NodeBase.ROOT, excludedNodes, blocksize,
225              maxNodesPerRack, results, avoidStaleNodes, storageTypes);
226        }
227      } else {
228        //otherwise randomly choose one from the network
229        return chooseRandom(NodeBase.ROOT, excludedNodes, blocksize,
230            maxNodesPerRack, results, avoidStaleNodes, storageTypes);
231      }
232    }
233  }

235  @Override
236  protected String getRack(final DatanodeInfo cur) {
237    String nodeGroupString = cur.getNetworkLocation();
238    return NetworkTopology.getFirstHalf(nodeGroupString);
239  }

241  /**
242   * Find other nodes in the same nodegroup of <i>localMachine</i> and add them
243   * into <i>excludeNodes</i> as replica should not be duplicated for nodes 
244   * within the same nodegroup
245   * @return number of new excluded nodes
246   */
247  @Override
248  protected int addToExcludedNodes(DatanodeDescriptor chosenNode,
249      Set<Node> excludedNodes) {
250    int countOfExcludedNodes = 0;
251    String nodeGroupScope = chosenNode.getNetworkLocation();
252    List<Node> leafNodes = clusterMap.getLeaves(nodeGroupScope);
253    for (Node leafNode : leafNodes) {
254      if (excludedNodes.add(leafNode)) {
255        // not a existing node in excludedNodes
256        countOfExcludedNodes++;
257      }
258    }
259    
260    countOfExcludedNodes += addDependentNodesToExcludedNodes(
261        chosenNode, excludedNodes);
262    return countOfExcludedNodes;
263  }
264  
265  /**
266   * Add all nodes from a dependent nodes list to excludedNodes.
267   * @return number of new excluded nodes
268   */
269  private int addDependentNodesToExcludedNodes(DatanodeDescriptor chosenNode,
270      Set<Node> excludedNodes) {
271    if (this.host2datanodeMap == null) {
272      return 0;
273    }
274    int countOfExcludedNodes = 0;
275    for(String hostname : chosenNode.getDependentHostNames()) {
276      DatanodeDescriptor node =
277          this.host2datanodeMap.getDataNodeByHostName(hostname);
278      if(node!=null) {
279        if (excludedNodes.add(node)) {
280          countOfExcludedNodes++;
281        }
282      } else {
283        LOG.warn("Not able to find datanode " + hostname
284            + " which has dependency with datanode "
285            + chosenNode.getHostName());
286      }
287    }
288    
289    return countOfExcludedNodes;
290  }

292  /**
293   * Pick up replica node set for deleting replica as over-replicated. 
294   * First set contains replica nodes on rack with more than one
295   * replica while second set contains remaining replica nodes.
296   * If first is not empty, divide first set into two subsets:
297   *   moreThanOne contains nodes on nodegroup with more than one replica
298   *   exactlyOne contains the remaining nodes in first set
299   * then pickup priSet if not empty.
300   * If first is empty, then pick second.
301   */
302  @Override
303  public Collection<DatanodeStorageInfo> pickupReplicaSet(
304      Collection<DatanodeStorageInfo> first,
305      Collection<DatanodeStorageInfo> second) {
306    // If no replica within same rack, return directly.
307    if (first.isEmpty()) {
308      return second;
309    }
310    // Split data nodes in the first set into two sets, 
311    // moreThanOne contains nodes on nodegroup with more than one replica
312    // exactlyOne contains the remaining nodes
313    Map<String, List<DatanodeStorageInfo>> nodeGroupMap = 
314        new HashMap<String, List<DatanodeStorageInfo>>();
315    
316    for(DatanodeStorageInfo storage : first) {
317      final String nodeGroupName = NetworkTopology.getLastHalf(
318          storage.getDatanodeDescriptor().getNetworkLocation());
319      List<DatanodeStorageInfo> storageList = nodeGroupMap.get(nodeGroupName);
320      if (storageList == null) {
321        storageList = new ArrayList<DatanodeStorageInfo>();
322        nodeGroupMap.put(nodeGroupName, storageList);
323      }
324      storageList.add(storage);
325    }
326    
327    final List<DatanodeStorageInfo> moreThanOne = new ArrayList<DatanodeStorageInfo>();
328    final List<DatanodeStorageInfo> exactlyOne = new ArrayList<DatanodeStorageInfo>();
329    // split nodes into two sets
330    for(List<DatanodeStorageInfo> datanodeList : nodeGroupMap.values()) {
331      if (datanodeList.size() == 1 ) {
332        // exactlyOne contains nodes on nodegroup with exactly one replica
333        exactlyOne.add(datanodeList.get(0));
334      } else {
335        // moreThanOne contains nodes on nodegroup with more than one replica
336        moreThanOne.addAll(datanodeList);
337      }
338    }
339    
340    return moreThanOne.isEmpty()? exactlyOne : moreThanOne;
341  }
