059  // Datanode administrative states
060  public enum AdminStates {
061    NORMAL("In Service"), 
062    DECOMMISSION_INPROGRESS("Decommission In Progress"), 
063    DECOMMISSIONED("Decommissioned");
064
065    final String value;
066
067    AdminStates(final String v) {
068      this.value = v;
069    }
070
071    @Override
072    public String toString() {
073      return value;
074    }
075    
076    public static AdminStates fromValue(final String value) {
077      for (AdminStates as : AdminStates.values()) {
078        if (as.value.equals(value)) return as;
079      }
080      return NORMAL;
081    }
082  }

319  /** A formatted string for reporting the status of the DataNode. */
320  public String getDatanodeReport() {
321    StringBuilder buffer = new StringBuilder();
322    long c = getCapacity();
323    long r = getRemaining();
324    long u = getDfsUsed();
325    long nonDFSUsed = getNonDfsUsed();
326    float usedPercent = getDfsUsedPercent();
327    float remainingPercent = getRemainingPercent();
328    long cc = getCacheCapacity();
329    long cr = getCacheRemaining();
330    long cu = getCacheUsed();
331    float cacheUsedPercent = getCacheUsedPercent();
332    float cacheRemainingPercent = getCacheRemainingPercent();
333    String lookupName = NetUtils.getHostNameOfIP(getName());
334
335    buffer.append("Name: "+ getName());
336    if (lookupName != null) {
337      buffer.append(" (" + lookupName + ")");
338    }
339    buffer.append("\n");
340    buffer.append("Hostname: " + getHostName() + "\n");
341
342    if (!NetworkTopology.DEFAULT_RACK.equals(location)) {
343      buffer.append("Rack: "+location+"\n");
344    }
345    buffer.append("Decommission Status : ");
346    if (isDecommissioned()) {
347      buffer.append("Decommissioned\n");
348    } else if (isDecommissionInProgress()) {
349      buffer.append("Decommission in progress\n");
350    } else {
351      buffer.append("Normal\n");
352    }
353    buffer.append("Configured Capacity: "+c+" ("+StringUtils.byteDesc(c)+")"+"\n");
354    buffer.append("DFS Used: "+u+" ("+StringUtils.byteDesc(u)+")"+"\n");
355    buffer.append("Non DFS Used: "+nonDFSUsed+" ("+StringUtils.byteDesc(nonDFSUsed)+")"+"\n");
356    buffer.append("DFS Remaining: " +r+ " ("+StringUtils.byteDesc(r)+")"+"\n");
357    buffer.append("DFS Used%: "+percent2String(usedPercent) + "\n");
358    buffer.append("DFS Remaining%: "+percent2String(remainingPercent) + "\n");
359    buffer.append("Configured Cache Capacity: "+cc+" ("+StringUtils.byteDesc(cc)+")"+"\n");
360    buffer.append("Cache Used: "+cu+" ("+StringUtils.byteDesc(cu)+")"+"\n");
361    buffer.append("Cache Remaining: " +cr+ " ("+StringUtils.byteDesc(cr)+")"+"\n");
362    buffer.append("Cache Used%: "+percent2String(cacheUsedPercent) + "\n");
363    buffer.append("Cache Remaining%: "+percent2String(cacheRemainingPercent) + "\n");
364    buffer.append("Xceivers: "+getXceiverCount()+"\n");
365    buffer.append("Last contact: "+new Date(lastUpdate)+"\n");
366    return buffer.toString();
367  }

369  /** A formatted string for printing the status of the DataNode. */
370  public String dumpDatanode() {
371    StringBuilder buffer = new StringBuilder();
372    long c = getCapacity();
373    long r = getRemaining();
374    long u = getDfsUsed();
375    long cc = getCacheCapacity();
376    long cr = getCacheRemaining();
377    long cu = getCacheUsed();
378    buffer.append(getName());
379    if (!NetworkTopology.DEFAULT_RACK.equals(location)) {
380      buffer.append(" "+location);
381    }
382    if (isDecommissioned()) {
383      buffer.append(" DD");
384    } else if (isDecommissionInProgress()) {
385      buffer.append(" DP");
386    } else {
387      buffer.append(" IN");
388    }
389    buffer.append(" " + c + "(" + StringUtils.byteDesc(c)+")");
390    buffer.append(" " + u + "(" + StringUtils.byteDesc(u)+")");
391    buffer.append(" " + percent2String(u/(double)c));
392    buffer.append(" " + r + "(" + StringUtils.byteDesc(r)+")");
393    buffer.append(" " + cc + "(" + StringUtils.byteDesc(cc)+")");
394    buffer.append(" " + cu + "(" + StringUtils.byteDesc(cu)+")");
395    buffer.append(" " + percent2String(cu/(double)cc));
396    buffer.append(" " + cr + "(" + StringUtils.byteDesc(cr)+")");
397    buffer.append(" " + new Date(lastUpdate));
398    return buffer.toString();
399  }

448  /**
449   * Check if the datanode is in stale state. Here if 
450   * the namenode has not received heartbeat msg from a 
451   * datanode for more than staleInterval (default value is
452   * {@link DFSConfigKeys#DFS_NAMENODE_STALE_DATANODE_INTERVAL_DEFAULT}),
453   * the datanode will be treated as stale node.
454   * 
455   * @param staleInterval
456   *          the time interval for marking the node as stale. If the last
457   *          update time is beyond the given time interval, the node will be
458   *          marked as stale.
459   * @return true if the node is stale
460   */
461  public boolean isStale(long staleInterval) {
462    return (Time.monotonicNow() - lastUpdateMonotonic) >= staleInterval;
463  }

466   * Sets the admin state of this node.
467   */
468  protected void setAdminState(AdminStates newState) {
469    if (newState == AdminStates.NORMAL) {
470      adminState = null;
471    }
472    else {
473      adminState = newState;
474    }
