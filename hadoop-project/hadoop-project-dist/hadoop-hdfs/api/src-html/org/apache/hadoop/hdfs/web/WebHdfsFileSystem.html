217  // the first getAuthParams() for a non-token op will either get the
218  // internal token from the ugi or lazy fetch one
219  protected synchronized Token<?> getDelegationToken() throws IOException {
220    if (canRefreshDelegationToken && delegationToken == null) {
221      Token<?> token = tokenSelector.selectToken(
222          new Text(getCanonicalServiceName()), ugi.getTokens());
223      // ugi tokens are usually indicative of a task which can't
224      // refetch tokens.  even if ugi has credentials, don't attempt
225      // to get another token to match hdfs/rpc behavior
226      if (token != null) {
227        LOG.debug("Using UGI token: " + token);
228        canRefreshDelegationToken = false; 
229      } else {
230        token = getDelegationToken(null);
231        if (token != null) {
232          LOG.debug("Fetched new token: " + token);
233        } else { // security is disabled
234          canRefreshDelegationToken = false;
235        }
236      }
237      setDelegationToken(token);
238    }
239    return delegationToken;
240  }

242  @VisibleForTesting
243  synchronized boolean replaceExpiredDelegationToken() throws IOException {
244    boolean replaced = false;
245    if (canRefreshDelegationToken) {
246      Token<?> token = getDelegationToken(null);
247      LOG.debug("Replaced expired token: " + token);
248      setDelegationToken(token);
249      replaced = (token != null);
250    }
251    return replaced;
252  }

298  static Map<?, ?> jsonParse(final HttpURLConnection c, final boolean useErrorStream
299      ) throws IOException {
300    if (c.getContentLength() == 0) {
301      return null;
302    }
303    final InputStream in = useErrorStream? c.getErrorStream(): c.getInputStream();
304    if (in == null) {
305      throw new IOException("The " + (useErrorStream? "error": "input") + " stream is null.");
306    }
307    try {
308      final String contentType = c.getContentType();
309      if (contentType != null) {
310        final MediaType parsed = MediaType.valueOf(contentType);
311        if (!MediaType.APPLICATION_JSON_TYPE.isCompatible(parsed)) {
312          throw new IOException("Content-Type \"" + contentType
313              + "\" is incompatible with \"" + MediaType.APPLICATION_JSON
314              + "\" (parsed=\"" + parsed + "\")");
315        }
316      }
317      ObjectMapper mapper = new ObjectMapper();
318      return mapper.reader(Map.class).readValue(in);
319    } finally {
320      in.close();
321    }
322  }

324  private static Map<?, ?> validateResponse(final HttpOpParam.Op op,
325      final HttpURLConnection conn, boolean unwrapException) throws IOException {
326    final int code = conn.getResponseCode();
327    // server is demanding an authentication we don't support
328    if (code == HttpURLConnection.HTTP_UNAUTHORIZED) {
329      // match hdfs/rpc exception
330      throw new AccessControlException(conn.getResponseMessage());
331    }
332    if (code != op.getExpectedHttpResponseCode()) {
333      final Map<?, ?> m;
334      try {
335        m = jsonParse(conn, true);
336      } catch(Exception e) {
337        throw new IOException("Unexpected HTTP response: code=" + code + " != "
338            + op.getExpectedHttpResponseCode() + ", " + op.toQueryString()
339            + ", message=" + conn.getResponseMessage(), e);
340      }
341
342      if (m == null) {
343        throw new IOException("Unexpected HTTP response: code=" + code + " != "
344            + op.getExpectedHttpResponseCode() + ", " + op.toQueryString()
345            + ", message=" + conn.getResponseMessage());
346      } else if (m.get(RemoteException.class.getSimpleName()) == null) {
347        return m;
348      }
349
350      IOException re = JsonUtil.toRemoteException(m);
351      // extract UGI-related exceptions and unwrap InvalidToken
352      // the NN mangles these exceptions but the DN does not and may need
353      // to re-fetch a token if either report the token is expired
354      if (re.getMessage() != null && re.getMessage().startsWith(
355          SecurityUtil.FAILED_TO_GET_UGI_MSG_HEADER)) {
356        String[] parts = re.getMessage().split(":\\s+", 3);
357        re = new RemoteException(parts[1], parts[2]);
358        re = ((RemoteException)re).unwrapRemoteException(InvalidToken.class);
359      }
360      throw unwrapException? toIOException(re): re;
361    }
362    return null;
363  }

365  /**
366   * Covert an exception to an IOException.
367   * 
368   * For a non-IOException, wrap it with IOException.
369   * For a RemoteException, unwrap it.
370   * For an IOException which is not a RemoteException, return it. 
371   */
372  private static IOException toIOException(Exception e) {
373    if (!(e instanceof IOException)) {
374      return new IOException(e);
375    }
376
377    final IOException ioe = (IOException)e;
378    if (!(ioe instanceof RemoteException)) {
379      return ioe;
380    }
381
382    return ((RemoteException)ioe).unwrapRemoteException();
383  }

414  Param<?,?>[] getAuthParameters(final HttpOpParam.Op op) throws IOException {
415    List<Param<?,?>> authParams = Lists.newArrayList();    
416    // Skip adding delegation token for token operations because these
417    // operations require authentication.
418    Token<?> token = null;
419    if (!op.getRequireAuth()) {
420      token = getDelegationToken();
421    }
422    if (token != null) {
423      authParams.add(new DelegationParam(token.encodeToUrlString()));
424    } else {
425      UserGroupInformation userUgi = ugi;
426      UserGroupInformation realUgi = userUgi.getRealUser();
427      if (realUgi != null) { // proxy user
428        authParams.add(new DoAsParam(userUgi.getShortUserName()));
429        userUgi = realUgi;
430      }
431      authParams.add(new UserParam(userUgi.getShortUserName()));
432    }
433    return authParams.toArray(new Param<?,?>[0]);
434  }

436  URL toUrl(final HttpOpParam.Op op, final Path fspath,
437      final Param<?,?>... parameters) throws IOException {
438    //initialize URI path and query
439    final String path = PATH_PREFIX
440        + (fspath == null? "/": makeQualified(fspath).toUri().getRawPath());
441    final String query = op.toQueryString()
442        + Param.toSortedString("&", getAuthParameters(op))
443        + Param.toSortedString("&", parameters);
444    final URL url = getNamenodeURL(path, query);
445    if (LOG.isTraceEnabled()) {
446      LOG.trace("url=" + url);
447    }
448    return url;
449  }

451  /**
452   * This class is for initialing a HTTP connection, connecting to server,
453   * obtaining a response, and also handling retry on failures.
454   */
455  abstract class AbstractRunner<T> {
456    abstract protected URL getUrl() throws IOException;
457
458    protected final HttpOpParam.Op op;
459    private final boolean redirected;
460    protected ExcludeDatanodesParam excludeDatanodes = new ExcludeDatanodesParam("");
461
462    private boolean checkRetry;
463
464    protected AbstractRunner(final HttpOpParam.Op op, boolean redirected) {
465      this.op = op;
466      this.redirected = redirected;
467    }
468
469    T run() throws IOException {
470      UserGroupInformation connectUgi = ugi.getRealUser();
471      if (connectUgi == null) {
472        connectUgi = ugi;
473      }
474      if (op.getRequireAuth()) {
475        connectUgi.checkTGTAndReloginFromKeytab();
476      }
477      try {
478        // the entire lifecycle of the connection must be run inside the
479        // doAs to ensure authentication is performed correctly
480        return connectUgi.doAs(
481            new PrivilegedExceptionAction<T>() {
482              @Override
483              public T run() throws IOException {
484                return runWithRetry();
485              }
486            });
487      } catch (InterruptedException e) {
488        throw new IOException(e);
489      }
490    }
491
492    /**
493     * Two-step requests redirected to a DN
494     * 
495     * Create/Append:
496     * Step 1) Submit a Http request with neither auto-redirect nor data. 
497     * Step 2) Submit another Http request with the URL from the Location header with data.
498     * 
499     * The reason of having two-step create/append is for preventing clients to
500     * send out the data before the redirect. This issue is addressed by the
501     * "Expect: 100-continue" header in HTTP/1.1; see RFC 2616, Section 8.2.3.
502     * Unfortunately, there are software library bugs (e.g. Jetty 6 http server
503     * and Java 6 http client), which do not correctly implement "Expect:
504     * 100-continue". The two-step create/append is a temporary workaround for
505     * the software library bugs.
506     * 
507     * Open/Checksum
508     * Also implements two-step connects for other operations redirected to
509     * a DN such as open and checksum
510     */
511    private HttpURLConnection connect(URL url) throws IOException {
512      //redirect hostname and port
513      String redirectHost = null;
514
515      
516      // resolve redirects for a DN operation unless already resolved
517      if (op.getRedirect() && !redirected) {
518        final HttpOpParam.Op redirectOp =
519            HttpOpParam.TemporaryRedirectOp.valueOf(op);
520        final HttpURLConnection conn = connect(redirectOp, url);
521        // application level proxy like httpfs might not issue a redirect
522        if (conn.getResponseCode() == op.getExpectedHttpResponseCode()) {
523          return conn;
524        }
525        try {
526          validateResponse(redirectOp, conn, false);
527          url = new URL(conn.getHeaderField("Location"));
528          redirectHost = url.getHost() + ":" + url.getPort();
529        } finally {
530          conn.disconnect();
531        }
532      }
533      try {
534        return connect(op, url);
535      } catch (IOException ioe) {
536        if (redirectHost != null) {
537          if (excludeDatanodes.getValue() != null) {
538            excludeDatanodes = new ExcludeDatanodesParam(redirectHost + ","
539                + excludeDatanodes.getValue());
540          } else {
541            excludeDatanodes = new ExcludeDatanodesParam(redirectHost);
542          }
543        }
544        throw ioe;
545      }      
546    }
547
548    private HttpURLConnection connect(final HttpOpParam.Op op, final URL url)
549        throws IOException {
550      final HttpURLConnection conn =
551          (HttpURLConnection)connectionFactory.openConnection(url);
552      final boolean doOutput = op.getDoOutput();
553      conn.setRequestMethod(op.getType().toString());
554      conn.setInstanceFollowRedirects(false);
555      switch (op.getType()) {
556        // if not sending a message body for a POST or PUT operation, need
557        // to ensure the server/proxy knows this 
558        case POST:
559        case PUT: {
560          conn.setDoOutput(true);
561          if (!doOutput) {
562            // explicitly setting content-length to 0 won't do spnego!!
563            // opening and closing the stream will send "Content-Length: 0"
564            conn.getOutputStream().close();
565          } else {
566            conn.setRequestProperty("Content-Type",
567                MediaType.APPLICATION_OCTET_STREAM);
568            conn.setChunkedStreamingMode(32 << 10); //32kB-chunk
569          }
570          break;
571        }
572        default: {
573          conn.setDoOutput(doOutput);
574          break;
575        }
576      }
577      conn.connect();
578      return conn;
579    }

581    private T runWithRetry() throws IOException {
582      /**
583       * Do the real work.
584       *
585       * There are three cases that the code inside the loop can throw an
586       * IOException:
587       *
588       * <ul>
589       * <li>The connection has failed (e.g., ConnectException,
590       * @see FailoverOnNetworkExceptionRetry for more details)</li>
591       * <li>The namenode enters the standby state (i.e., StandbyException).</li>
592       * <li>The server returns errors for the command (i.e., RemoteException)</li>
593       * </ul>
594       *
595       * The call to shouldRetry() will conduct the retry policy. The policy
596       * examines the exception and swallows it if it decides to rerun the work.
597       */
598      for(int retry = 0; ; retry++) {
599        checkRetry = !redirected;
600        final URL url = getUrl();
601        try {
602          final HttpURLConnection conn = connect(url);
603          // output streams will validate on close
604          if (!op.getDoOutput()) {
605            validateResponse(op, conn, false);
606          }
607          return getResponse(conn);
608        } catch (AccessControlException ace) {
609          // no retries for auth failures
610          throw ace;
611        } catch (InvalidToken it) {
612          // try to replace the expired token with a new one.  the attempt
613          // to acquire a new token must be outside this operation's retry
614          // so if it fails after its own retries, this operation fails too.
615          if (op.getRequireAuth() || !replaceExpiredDelegationToken()) {
616            throw it;
617          }
618        } catch (IOException ioe) {
619          shouldRetry(ioe, retry);
620        }
621      }
622    }
623
624    private void shouldRetry(final IOException ioe, final int retry
625        ) throws IOException {
626      InetSocketAddress nnAddr = getCurrentNNAddr();
627      if (checkRetry) {
628        try {
629          final RetryPolicy.RetryAction a = retryPolicy.shouldRetry(
630              ioe, retry, 0, true);
631
632          boolean isRetry = a.action == RetryPolicy.RetryAction.RetryDecision.RETRY;
633          boolean isFailoverAndRetry =
634              a.action == RetryPolicy.RetryAction.RetryDecision.FAILOVER_AND_RETRY;
635
636          if (isRetry || isFailoverAndRetry) {
637            LOG.info("Retrying connect to namenode: " + nnAddr
638                + ". Already tried " + retry + " time(s); retry policy is "
639                + retryPolicy + ", delay " + a.delayMillis + "ms.");
640
641            if (isFailoverAndRetry) {
642              resetStateToFailOver();
643            }
644
645            Thread.sleep(a.delayMillis);
646            return;
647          }
648        } catch(Exception e) {
649          LOG.warn("Original exception is ", ioe);
650          throw toIOException(e);
651        }
652      }
653      throw toIOException(ioe);
654    }

659  /**
660   * Abstract base class to handle path-based operations with params
661   */
662  abstract class AbstractFsPathRunner<T> extends AbstractRunner<T> {
663    private final Path fspath;
664    private final Param<?,?>[] parameters;
665    
666    AbstractFsPathRunner(final HttpOpParam.Op op, final Path fspath,
667        Param<?,?>... parameters) {
668      super(op, false);
669      this.fspath = fspath;
670      this.parameters = parameters;
671    }
672    
673    AbstractFsPathRunner(final HttpOpParam.Op op, Param<?,?>[] parameters,
674        final Path fspath) {
675      super(op, false);
676      this.fspath = fspath;
677      this.parameters = parameters;
678    }
679    
680    @Override
681    protected URL getUrl() throws IOException {
682      if (excludeDatanodes.getValue() != null) {
683        Param<?, ?>[] tmpParam = new Param<?, ?>[parameters.length + 1];
684        System.arraycopy(parameters, 0, tmpParam, 0, parameters.length);
685        tmpParam[parameters.length] = excludeDatanodes;
686        return toUrl(op, fspath, tmpParam);
687      } else {
688        return toUrl(op, fspath, parameters);
689      }
690    }
691  }

707  /**
708   * Handle path-based operations with a json response
709   */
710  abstract class FsPathResponseRunner<T> extends AbstractFsPathRunner<T> {
711    FsPathResponseRunner(final HttpOpParam.Op op, final Path fspath,
712        Param<?,?>... parameters) {
713      super(op, fspath, parameters);
714    }
715    
716    FsPathResponseRunner(final HttpOpParam.Op op, Param<?,?>[] parameters,
717        final Path fspath) {
718      super(op, parameters, fspath);
719    }
720    
721    @Override
722    final T getResponse(HttpURLConnection conn) throws IOException {
723      try {
724        final Map<?,?> json = jsonParse(conn, false);
725        if (json == null) {
726          // match exception class thrown by parser
727          throw new IllegalStateException("Missing response");
728        }
729        return decodeResponse(json);
730      } catch (IOException ioe) {
731        throw ioe;
732      } catch (Exception e) { // catch json parser errors
733        final IOException ioe =
734            new IOException("Response decoding failure: "+e.toString(), e);
735        if (LOG.isDebugEnabled()) {
736          LOG.debug(ioe);
737        }
738        throw ioe;
739      } finally {
740        conn.disconnect();
741      }
742    }
743    
744    abstract T decodeResponse(Map<?,?> json) throws IOException;
745  }

1230  private static final String OFFSET_PARAM_PREFIX = OffsetParam.NAME + "=";
1231
1232  /** Remove offset parameter, if there is any, from the url */
1233  static URL removeOffsetParam(final URL url) throws MalformedURLException {
1234    String query = url.getQuery();
1235    if (query == null) {
1236      return url;
1237    }
1238    final String lower = StringUtils.toLowerCase(query);
1239    if (!lower.startsWith(OFFSET_PARAM_PREFIX)
1240        && !lower.contains("&" + OFFSET_PARAM_PREFIX)) {
1241      return url;
1242    }
1243
1244    //rebuild query
1245    StringBuilder b = null;
1246    for(final StringTokenizer st = new StringTokenizer(query, "&");
1247        st.hasMoreTokens();) {
1248      final String token = st.nextToken();
1249      if (!StringUtils.toLowerCase(token).startsWith(OFFSET_PARAM_PREFIX)) {
1250        if (b == null) {
1251          b = new StringBuilder("?").append(token);
1252        } else {
1253          b.append('&').append(token);
1254        }
1255      }
1256    }
1257    query = b == null? "": b.toString();
1258
1259    final String urlStr = url.toString();
1260    return new URL(urlStr.substring(0, urlStr.indexOf('?')) + query);
1261  }

1417  /**
1418   * Resolve an HDFS URL into real INetSocketAddress. It works like a DNS
1419   * resolver when the URL points to an non-HA cluster. When the URL points to
1420   * an HA cluster with its logical name, the resolver further resolves the
1421   * logical name(i.e., the authority in the URL) into real namenode addresses.
1422   */
1423  private InetSocketAddress[] resolveNNAddr() throws IOException {
1424    Configuration conf = getConf();
1425    final String scheme = uri.getScheme();
1426
1427    ArrayList<InetSocketAddress> ret = new ArrayList<InetSocketAddress>();
1428
1429    if (!HAUtil.isLogicalUri(conf, uri)) {
1430      InetSocketAddress addr = NetUtils.createSocketAddr(uri.getAuthority(),
1431          getDefaultPort());
1432      ret.add(addr);
1433
1434    } else {
1435      Map<String, Map<String, InetSocketAddress>> addresses = DFSUtil
1436          .getHaNnWebHdfsAddresses(conf, scheme);
1437
1438      // Extract the entry corresponding to the logical name.
1439      Map<String, InetSocketAddress> addrs = addresses.get(uri.getHost());
1440      for (InetSocketAddress addr : addrs.values()) {
1441        ret.add(addr);
1442      }
1443    }
1444
1445    InetSocketAddress[] r = new InetSocketAddress[ret.size()];
1446    return ret.toArray(r);
1447  }
