199  //This method enables handling UGI doAs with SPNEGO, we have to
200  //fallback to the realuser who logged in with Kerberos credentials
201  private <T> T doAsRealUserIfNecessary(final Callable<T> callable)
202    throws IOException {
203    try {
204      if (realUser.getShortUserName().equals(doAs)) {
205        return callable.call();
206      } else {
207        return realUser.doAs(new PrivilegedExceptionAction<T>() {
208          @Override
209          public T run() throws Exception {
210            return callable.call();
211          }
212        });
213      }
214    } catch (Exception ex) {
215      throw new IOException(ex.toString(), ex);
216    }
217  }

219  /**
220   * Convenience method that creates a <code>HttpURLConnection</code> for the
221   * HttpFSServer file system operations.
222   * <p/>
223   * This methods performs and injects any needed authentication credentials
224   * via the {@link #getConnection(URL, String)} method
225   *
226   * @param method the HTTP method.
227   * @param params the query string parameters.
228   * @param path the file path
229   * @param makeQualified if the path should be 'makeQualified'
230   *
231   * @return a <code>HttpURLConnection</code> for the HttpFSServer server,
232   *         authenticated and ready to use for the specified path and file system operation.
233   *
234   * @throws IOException thrown if an IO error occurrs.
235   */
236  private HttpURLConnection getConnection(final String method,
237      Map<String, String> params, Path path, boolean makeQualified)
238      throws IOException {
239    if (!realUser.getShortUserName().equals(doAs)) {
240      params.put(DO_AS_PARAM, doAs);
241    }
242    HttpFSKerberosAuthenticator.injectDelegationToken(params, delegationToken);
243    if (makeQualified) {
244      path = makeQualified(path);
245    }
246    final URL url = HttpFSUtils.createURL(path, params);
247    return doAsRealUserIfNecessary(new Callable<HttpURLConnection>() {
248      @Override
249      public HttpURLConnection call() throws Exception {
250        return getConnection(url, method);
251      }
252    });
253  }

255  /**
256   * Convenience method that creates a <code>HttpURLConnection</code> for the specified URL.
257   * <p/>
258   * This methods performs and injects any needed authentication credentials.
259   *
260   * @param url url to connect to.
261   * @param method the HTTP method.
262   *
263   * @return a <code>HttpURLConnection</code> for the HttpFSServer server, authenticated and ready to use for
264   *         the specified path and file system operation.
265   *
266   * @throws IOException thrown if an IO error occurrs.
267   */
268  private HttpURLConnection getConnection(URL url, String method) throws IOException {
269    Class<? extends Authenticator> klass =
270      getConf().getClass("httpfs.authenticator.class",
271                         HttpFSKerberosAuthenticator.class, Authenticator.class);
272    Authenticator authenticator = ReflectionUtils.newInstance(klass, getConf());
273    try {
274      HttpURLConnection conn = new AuthenticatedURL(authenticator).openConnection(url, authToken);
275      conn.setRequestMethod(method);
276      if (method.equals(HTTP_POST) || method.equals(HTTP_PUT)) {
277        conn.setDoOutput(true);
278      }
279      return conn;
280    } catch (Exception ex) {
281      throw new IOException(ex);
282    }
283  }

285  /**
286   * Called after a new FileSystem instance is constructed.
287   *
288   * @param name a uri whose authority section names the host, port, etc. for this FileSystem
289   * @param conf the configuration
290   */
291  @Override
292  public void initialize(URI name, Configuration conf) throws IOException {
293    UserGroupInformation ugi = UserGroupInformation.getCurrentUser();
294
295    //the real use is the one that has the Kerberos credentials needed for
296    //SPNEGO to work
297    realUser = ugi.getRealUser();
298    if (realUser == null) {
299      realUser = UserGroupInformation.getLoginUser();
300    }
301    doAs = ugi.getShortUserName();
302    super.initialize(name, conf);
303    try {
304      uri = new URI(name.getScheme() + "://" + name.getAuthority());
305      httpFSAddr = NetUtils.createSocketAddr(getCanonicalUri().toString());
306    } catch (URISyntaxException ex) {
307      throw new IOException(ex);
308    }
309  }

437  /*
438   * Common handling for uploading data for create and append operations.
439   */
440  private FSDataOutputStream uploadData(String method, Path f, Map<String, String> params,
441                                        int bufferSize, int expectedStatus) throws IOException {
442    HttpURLConnection conn = getConnection(method, params, f, true);
443    conn.setInstanceFollowRedirects(false);
444    boolean exceptionAlreadyHandled = false;
445    try {
446      if (conn.getResponseCode() == HTTP_TEMPORARY_REDIRECT) {
447        exceptionAlreadyHandled = true;
448        String location = conn.getHeaderField("Location");
449        if (location != null) {
450          conn = getConnection(new URL(location), method);
451          conn.setRequestProperty("Content-Type", UPLOAD_CONTENT_TYPE);
452          try {
453            OutputStream os = new BufferedOutputStream(conn.getOutputStream(), bufferSize);
454            return new HttpFSDataOutputStream(conn, os, expectedStatus, statistics);
455          } catch (IOException ex) {
456            HttpFSUtils.validateResponse(conn, expectedStatus);
457            throw ex;
458          }
459        } else {
460          HttpFSUtils.validateResponse(conn, HTTP_TEMPORARY_REDIRECT);
461          throw new IOException("Missing HTTP 'Location' header for [" + conn.getURL() + "]");
462        }
463      } else {
464        throw new IOException(
465          MessageFormat.format("Expected HTTP status was [307], received [{0}]",
466                               conn.getResponseCode()));
467      }
468    } catch (IOException ex) {
469      if (exceptionAlreadyHandled) {
470        throw ex;
471      } else {
472        HttpFSUtils.validateResponse(conn, HTTP_TEMPORARY_REDIRECT);
473        throw ex;
474      }
475    }
476  }

533  /**
534   * Concat existing files together.
535   * @param f the path to the target destination.
536   * @param psrcs the paths to the sources to use for the concatenation.
537   *
538   * @throws IOException
539   */
540  @Override
541  public void concat(Path f, Path[] psrcs) throws IOException {
542    List<String> strPaths = new ArrayList<String>(psrcs.length);
543    for(Path psrc : psrcs) {
544      strPaths.add(psrc.toUri().getPath());
545    }
546    String srcs = StringUtils.join(",", strPaths);
547
548    Map<String, String> params = new HashMap<String, String>();
549    params.put(OP_PARAM, Operation.CONCAT.toString());
550    params.put(SOURCES_PARAM, srcs);
551    HttpURLConnection conn = getConnection(Operation.CONCAT.getMethod(),
552        params, f, true);
553    HttpFSUtils.validateResponse(conn, HttpURLConnection.HTTP_OK);
554  }

692                                           params, f, true);

696    f = makeQualified(f);

801  private FileStatus createFileStatus(Path parent, JSONObject json) {
802    String pathSuffix = (String) json.get(PATH_SUFFIX_JSON);
803    Path path = (pathSuffix.equals("")) ? parent : new Path(parent, pathSuffix);
804    FILE_TYPE type = FILE_TYPE.valueOf((String) json.get(TYPE_JSON));
805    long len = (Long) json.get(LENGTH_JSON);
806    String owner = (String) json.get(OWNER_JSON);
807    String group = (String) json.get(GROUP_JSON);
808    FsPermission permission =
809      new FsPermission(Short.parseShort((String) json.get(PERMISSION_JSON), 8));
810    long aTime = (Long) json.get(ACCESS_TIME_JSON);
811    long mTime = (Long) json.get(MODIFICATION_TIME_JSON);
812    long blockSize = (Long) json.get(BLOCK_SIZE_JSON);
813    short replication = ((Long) json.get(REPLICATION_JSON)).shortValue();
814    FileStatus fileStatus = null;
815
816    switch (type) {
817      case FILE:
818      case DIRECTORY:
819        fileStatus = new FileStatus(len, (type == FILE_TYPE.DIRECTORY),
820                                    replication, blockSize, mTime, aTime,
821                                    permission, owner, group, path);
822        break;
823      case SYMLINK:
824        Path symLink = null;
825        fileStatus = new FileStatus(len, false,
826                                    replication, blockSize, mTime, aTime,
827                                    permission, owner, group, symLink,
828                                    path);
829    }
830    return fileStatus;
831  }
